34  private void onGraspedMovement(Vector3 presolvePos, Quaternion presolveRot,
35  Vector3 solvedPos, Quaternion solvedRot,
36  List<InteractionController> controllers) {
37  /*
38  * OnGraspedMovement provides the position and rotation of the Interaction object
39  * before and after it was moved by its grasping hand. This callback only occurs
40  * when one or more hands is grasping the Interaction object. In this case, we
41  * don't care about how many or which hands are grasping the object, only where
42  * the object is moved.
43  *
44  * The Translation Handle uses the pre- and post-solve movement information to
45  * calculate how the user is trying to move the object along this handle's forward
46  * direction. Then the Translation Handle will simply override the movement caused
47  * by the grasping hand and reset itself back to its original position.
48  *
49  * The movement calculated by the Handle in this method is reported to the Transform
50  * Tool, which accumulates movement caused by all Handles over the course of a frame
51  * and then moves the target object and all of its child Handles appropriately at
52  * the end of the frame.
53  */
54 
55  // Calculate the constrained movement of the handle along its forward axis only.
56  Vector3 deltaPos = solvedPos - presolvePos;
57  Vector3 handleForwardDirection = presolveRot * Vector3.forward;
58  Vector3 deltaAxisPos = handleForwardDirection * Vector3.Dot(handleForwardDirection, deltaPos);
59 
60  // Notify the tool about the calculated movement.
61  _tool.NotifyHandleMovement(deltaAxisPos);
62 
63  // In this case, the tool itself will accumulate delta positions and delta rotations
64  // from all handles, and will then synchronize handles to the appropriate positions and
65  // rotations at the end of the frame.
66 
67  // Because the Tool will be the one to actually move this Handle, all we have left to do
68  // is to undo all of the motion caused by the grasping hand.
69  _intObj.rigidbody.position = presolvePos;
70  _intObj.rigidbody.rotation = presolveRot;
71  }
