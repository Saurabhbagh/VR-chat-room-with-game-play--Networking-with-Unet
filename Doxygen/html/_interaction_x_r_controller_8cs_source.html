322  private const float RIG_LOCAL_MOVEMENT_SPEED_THRESHOLD = 00.07F;
323  private const float RIG_LOCAL_MOVEMENT_SPEED_THRESHOLD_SQR
324  = RIG_LOCAL_MOVEMENT_SPEED_THRESHOLD * RIG_LOCAL_MOVEMENT_SPEED_THRESHOLD;
325  private const float RIG_LOCAL_ROTATION_SPEED_THRESHOLD = 10.00F;
326  private const float BEING_MOVED_TIMEOUT = 0.5F;
327 
328  private float _lastTimeMoved = 0F;
329  private bool _isBeingMoved = false;
330  private void refreshIsBeingMoved(Vector3 position, Quaternion rotation) {
331  var isMoving = false;
332  var baseTransform = this.manager.transform;
333 
334  // Check translation speed, relative to the Interaction Manager.
335  var baseLocalPos = baseTransform.InverseTransformPoint(position);
336  var baseLocalPosLastFrame =baseTransform.InverseTransformPoint(
337  _trackedPositionLastFrame);
338  var baseLocalSqrSpeed = ((baseLocalPos - baseLocalPosLastFrame)
339  / Time.fixedDeltaTime).sqrMagnitude;
340  if (baseLocalSqrSpeed > RIG_LOCAL_MOVEMENT_SPEED_THRESHOLD_SQR) {
341  isMoving = true;
342  }
343 
344  // Check rotation speed, relative to the Interaction Manager.
345  var baseLocalRot = baseTransform.InverseTransformRotation(rotation);
346  var baseLocalRotLastFrame = baseTransform.InverseTransformRotation(
347  _trackedRotationLastFrame);
348  var baseLocalAngularSpeed = Quaternion.Angle(baseLocalRot, baseLocalRotLastFrame)
349  / Time.fixedDeltaTime;
350  if (baseLocalAngularSpeed > RIG_LOCAL_ROTATION_SPEED_THRESHOLD) {
351  isMoving = true;
352  }
353 
354  if (isMoving) {
355  _lastTimeMoved = Time.fixedTime;
356  }
357 
358  // "isMoving" lasts for a bit after the controller stops moving, to avoid
359  // rapid oscillation of the value.
360  var timeSinceLastMoving = Time.fixedTime - _lastTimeMoved;
361  _isBeingMoved = trackingProvider != null && trackingProvider.isTracked
362  && timeSinceLastMoving < BEING_MOVED_TIMEOUT;
363  }

497  protected override void unwarpColliders(Transform primaryHoverPoint, ISpaceComponent warpedSpaceElement) {
498  // Extension method calculates "unwarped" pose in world space.
499  Vector3 unwarpedPosition;
500  Quaternion unwarpedRotation;

509  _pivotingPositionOffset = -primaryHoverPoint.position;

584  _colliderBuffer.Clear();

588  Utils.FindColliders<Collider>(this.gameObject, _colliderBuffer,
589  includeInactiveObjects: true);
590 
591  foreach (var collider in _colliderBuffer) {
592  if (collider.isTrigger) continue; // Contact Bones are for "contacting" colliders.
593

596  if (body == null) {

598  }

602  body.collisionDetectionMode = CollisionDetectionMode.ContinuousDynamic;
603  body.mass = 1F;
604 
605  contactBone.interactionController = this;
606  contactBone.rigidbody = body;
607  contactBone.collider = collider;
608 
609  _contactBoneBuffer.Add(contactBone);
610  }

617  _contactBoneTargetRotations = new Quaternion[numBones];

739  }

743  _graspButtonDown = true;

752  }

755  try {
756  graspButton = Input.GetAxis(graspButtonAxis) > graspDepressedValue;
757  } catch {

764  }
