413  {
414  AnimatorControllerParameter[] parameters = handAnimator.parameters;
415  for (int parameterIndex = 0; parameterIndex < parameters.Length; parameterIndex++)
416  {
417  if (string.Equals(parameters[parameterIndex].name, animatorParameterStateName, StringComparison.CurrentCultureIgnoreCase))
418  handAnimatorStateId = parameters[parameterIndex].nameHash;
419  }
420  }
