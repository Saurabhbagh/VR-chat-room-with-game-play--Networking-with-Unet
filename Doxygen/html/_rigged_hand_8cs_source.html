245  private Vector3 calculateModelPalmFacing(Transform palm, Transform finger1, Transform finger2) {
246  Vector3 a = palm.transform.InverseTransformPoint(palm.position);
247  Vector3 b = palm.transform.InverseTransformPoint(finger1.position);
248  Vector3 c = palm.transform.InverseTransformPoint(finger2.position);
249 
250  Vector3 side1 = b - a;
251  Vector3 side2 = c - a;
252  Vector3 perpendicular;
253 
254  if (Handedness == Chirality.Left) {
255  perpendicular = Vector3.Cross(side1, side2);
256  }
257  else perpendicular = Vector3.Cross(side2, side1);
258  //flip perpendicular if it is above palm
259  Vector3 calculatedPalmFacing = CalculateZeroedVector(perpendicular);
260  return calculatedPalmFacing;
261  }
263  private Vector3 calculateModelFingerPointing() {
264  Vector3 distance = palm.transform.InverseTransformPoint(fingers[2].transform.GetChild(0).transform.position) - palm.transform.InverseTransformPoint(palm.position);
265  Vector3 calculatedFingerPointing = CalculateZeroedVector(distance);
266  return calculatedFingerPointing * -1f;
267  }
269  public static Vector3 CalculateZeroedVector(Vector3 vectorToZero) {
270  var zeroed = new Vector3();
271  float max = Mathf.Max(Mathf.Abs(vectorToZero.x), Mathf.Abs(vectorToZero.y), Mathf.Abs(vectorToZero.z));
272  if (Mathf.Abs(vectorToZero.x) == max) {
273  zeroed = (vectorToZero.x < 0) ? new Vector3(1, 0, 0) : new Vector3(-1, 0, 0);
274  }
275  if (Mathf.Abs(vectorToZero.y) == max) {
276  zeroed = (vectorToZero.y < 0) ? new Vector3(0, 1, 0) : new Vector3(0, -1, 0);
277  }
278  if (Mathf.Abs(vectorToZero.z) == max) {
279  zeroed = (vectorToZero.z < 0) ? new Vector3(0, 0, 1) : new Vector3(0, 0, -1);
280  }
281  return zeroed;
282  }
