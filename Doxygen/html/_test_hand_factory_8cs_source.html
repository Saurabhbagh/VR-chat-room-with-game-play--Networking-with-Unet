180  private static LeapQuaternion angleAxis(float angle, Vector axis) {
181  if (!axis.MagnitudeSquared.NearlyEquals(1.0f)) {
182  throw new ArgumentException("Axis must be a unit vector.");
183  }
184  float sineHalfAngle = Mathf.Sin(angle/2.0f);
185  LeapQuaternion q = new LeapQuaternion(sineHalfAngle * axis.x,
186  sineHalfAngle * axis.y,
187  sineHalfAngle * axis.z,
188  Mathf.Cos(angle/2.0f));
189  return q.Normalized;
190  }

192  private static LeapQuaternion rotationBetween(Vector fromDirection, Vector toDirection) {

194  Vector w = (1.0f / m) * fromDirection.Cross(toDirection);
195  return new LeapQuaternion(w.x, w.y, w.z, 0.5f * m);
