213  public float GetTargetDistance()
214  {
215  if (target == null)
216  return config.near + 0.01f;
217 
218  var offset = cam.transform;
219  var forward = new Vector3(offset.forward.x, 0.0f, offset.forward.z).normalized;
220  var targetPos = target.position + new Vector3(target.forward.x, 0.0f, target.forward.z).normalized * config.hmdOffset;
221 
222  var distance = -(new Plane(forward, targetPos)).GetDistanceToPoint(offset.position);
223  return Mathf.Clamp(distance, config.near + 0.01f, config.far - 0.01f);
224  }
