223  protected virtual void FixedUpdate() {
224  if (!_physicsOccurred) {
225  _physicsOccurred = true;
226 
227  if (!isGrasped && !rigidbody.IsSleeping()) {
228 
229  float localPhysicsDisplacementPercentage
230  = Mathf.InverseLerp(minMaxHeight.x, minMaxHeight.y,
231  initialLocalPosition.z - localPhysicsPosition.z);
232 
233  // Sleep the rigidbody if it's not really moving.
234  if (rigidbody.position == physicsPosition
235  && _physicsVelocity == Vector3.zero
236  && Mathf.Abs(localPhysicsDisplacementPercentage - restingHeight) < 0.01F) {
237  rigidbody.Sleep();

239  // Otherwise reset the body's position to where it was last time PhysX

252  private const float FRICTION_COEFFICIENT = 30F;
253  private const float DRAG_COEFFICIENT = 60F;
254  protected virtual void Update() {
255 
256  // Reset our convenience state variables.
257  _pressedThisFrame = false;
258  _unpressedThisFrame = false;
259 
260  // Disable collision on this button if it is not the primary hover.
261  ignoreGrasping = _initialIgnoreGrasping ? true : !isPrimaryHovered && !isGrasped;
262  ignoreContact = (!isPrimaryHovered || isGrasped) || !controlEnabled;
263 
264  // Enforce local rotation (if button is child of non-kinematic rigidbody,
265  // this is necessary).
266  transform.localRotation = _initialLocalRotation;
267 
268  // Apply physical corrections only if PhysX has modified our positions.
269  if (_physicsOccurred) {
270  _physicsOccurred = false;
271 
272  // Record and enforce the sliding state from the previous frame.
273  if (this.primaryHoverDistance < 0.005f || isGrasped) {
274  localPhysicsPosition
275  = constrainDepressedLocalPosition(
276  transform.parent.InverseTransformPoint(rigidbody.position)
277  - localPhysicsPosition);
278  } else {
279  Vector2 localSlidePosition = new Vector2(localPhysicsPosition.x,
280  localPhysicsPosition.y);
281 
282  localPhysicsPosition
283  = transform.parent.InverseTransformPoint(this.transform.position);
284 
285  localPhysicsPosition = new Vector3(localSlidePosition.x,
286  localSlidePosition.y,
287  localPhysicsPosition.z);
288  }
289 
290  // Calculate the physical kinematics of the button in local space
291  Vector3 localPhysicsVelocity = transform.parent.InverseTransformVector(rigidbody.velocity);
292  if (isPressed && isPrimaryHovered && _lastDepressor != null) {
293  Vector3 curLocalDepressorPos = transform.parent.InverseTransformPoint(_lastDepressor.position);
294  Vector3 origLocalDepressorPos = transform.parent.InverseTransformPoint(transform.TransformPoint(_localDepressorPosition));
295  localPhysicsVelocity = Vector3.back * 0.05f;
296  localPhysicsPosition = constrainDepressedLocalPosition(curLocalDepressorPos - origLocalDepressorPos);
297  }
298  else if (isGrasped) {
299  // Do nothing!
300  }
301  else {
302  Vector3 originalLocalVelocity = localPhysicsVelocity;
303 
304  // Spring force
305  localPhysicsVelocity +=
306  Mathf.Clamp(_springForce * 10000F
307  * (initialLocalPosition.z
308  - Mathf.Lerp(minMaxHeight.x, minMaxHeight.y, restingHeight)
309  - localPhysicsPosition.z),
310  -100f / transform.parent.lossyScale.x,
311  100f / transform.parent.lossyScale.x)
312  * Time.fixedDeltaTime * Vector3.forward;
313 
314  // Friction & Drag
315  float velMag = originalLocalVelocity.magnitude;
316  var frictionDragVelocityChangeAmt = 0f;
317  if (velMag > 0F) {
318  // Friction force
319  var frictionForceAmt = velMag * FRICTION_COEFFICIENT;
320  frictionDragVelocityChangeAmt
321  += Time.fixedDeltaTime * transform.parent.lossyScale.x * frictionForceAmt;
322 
323  // Drag force
324  float velSqrMag = velMag * velMag;
325  var dragForceAmt = velSqrMag * DRAG_COEFFICIENT;
326  frictionDragVelocityChangeAmt
327  += Time.fixedDeltaTime * transform.parent.lossyScale.x * dragForceAmt;
328 
329  // Apply velocity change, but don't let friction or drag let velocity
330  // magnitude cross zero.
331  var newVelMag = Mathf.Max(0, velMag - frictionDragVelocityChangeAmt);
332  localPhysicsVelocity = localPhysicsVelocity / velMag * newVelMag;
333  }
334 
335  }
336 
337  // Transform the local physics back into world space
338  physicsPosition = transform.parent.TransformPoint(localPhysicsPosition);
339  _physicsVelocity = transform.parent.TransformVector(localPhysicsVelocity);
340 
341  // Calculate the Depression State of the Button from its Physical Position
342  // Set its Graphical Position to be Constrained Physically
343  bool oldDepressed = isPressed;
344 
345  // Normalized depression amount.
346  _pressedAmount = localPhysicsPosition.z.Map(initialLocalPosition.z - minMaxHeight.x,
347  initialLocalPosition.z - Mathf.Lerp(minMaxHeight.x, minMaxHeight.y, restingHeight),
348  1F, 0F);
349 
350  // If the button is depressed past its limit...
351  if (localPhysicsPosition.z > initialLocalPosition.z - minMaxHeight.x) {
352  transform.localPosition = new Vector3(localPhysicsPosition.x, localPhysicsPosition.y, initialLocalPosition.z - minMaxHeight.x);
353  if ((isPrimaryHovered && _lastDepressor != null) || isGrasped) {
354  _isPressed = true;
355  }
356  else {
357  physicsPosition = transform.parent.TransformPoint(new Vector3(localPhysicsPosition.x, localPhysicsPosition.y, initialLocalPosition.z - minMaxHeight.x));
358  _physicsVelocity = _physicsVelocity * 0.1f;
359  _isPressed = false;
360  _lastDepressor = null;
361  }
362  // Else if the button is extended past its limit...
363  }
364  else if (localPhysicsPosition.z < initialLocalPosition.z - minMaxHeight.y) {
365  transform.localPosition = new Vector3(localPhysicsPosition.x, localPhysicsPosition.y, initialLocalPosition.z - minMaxHeight.y);
366  physicsPosition = transform.position;
367  _isPressed = false;
368  _lastDepressor = null;
369  }
370  else {
371  // Else, just make the physical and graphical motion of the button match
372  transform.localPosition = localPhysicsPosition;
373 
374  // Allow some hysteresis before setting isDepressed to false.
375  if (!isPressed
376  || !(localPhysicsPosition.z > initialLocalPosition.z - (minMaxHeight.y - minMaxHeight.x) * 0.1F)) {
377  _isPressed = false;
378  _lastDepressor = null;
379  }
380  }
381 
382  // If our depression state has changed since last time...
383  if (isPressed && !oldDepressed) {
384  primaryHoveringController.primaryHoverLocked = true;
385  _lockedInteractingController = primaryHoveringController;
386 
387  OnPress();
388  _pressedThisFrame = true;
389 
390  } else if (!isPressed && oldDepressed) {
391  _unpressedThisFrame = true;
392  OnUnpress();
393 
394  if (!(isGrasped && graspingController == _lockedInteractingController)) {
395  _lockedInteractingController.primaryHoverLocked = false;
396  }
397 
398  _lastDepressor = null;
399  }
400  }
