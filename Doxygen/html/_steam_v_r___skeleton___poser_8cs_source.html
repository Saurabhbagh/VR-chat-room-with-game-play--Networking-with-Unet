194  public void UpdatePose(SteamVR_Action_Skeleton skeletonAction, SteamVR_Input_Sources inputSource)
195  {
196  // only allow this function to run once per frame
197  if (poseUpdatedThisFrame) return;
198 
199  poseUpdatedThisFrame = true;
200 
201  // always do additive animation on main pose
202  blendPoses[0].UpdateAdditiveAnimation(skeletonAction, inputSource);
203 
204  //copy from main pose as a base
205  SteamVR_Skeleton_PoseSnapshot snap = GetHandSnapshot(inputSource);
206  snap.CopyFrom(blendPoses[0].GetHandSnapshot(inputSource));
207 
208  ApplyBlenderBehaviours(skeletonAction, inputSource, snap);
209 
210 
211  if (inputSource == SteamVR_Input_Sources.RightHand)
212  blendedSnapshotR = snap;
213  if (inputSource == SteamVR_Input_Sources.LeftHand)
214  blendedSnapshotL = snap;
215  }

217  protected void ApplyBlenderBehaviours(SteamVR_Action_Skeleton skeletonAction, SteamVR_Input_Sources inputSource, SteamVR_Skeleton_PoseSnapshot snapshot)
218  {
219 
220  // apply blending for each behaviour
221  for (int behaviourIndex = 0; behaviourIndex < blendingBehaviours.Count; behaviourIndex++)
222  {
223  blendingBehaviours[behaviourIndex].Update(Time.deltaTime, inputSource);
224  // if disabled or very low influence, skip for perf
225  if (blendingBehaviours[behaviourIndex].enabled && blendingBehaviours[behaviourIndex].influence * blendingBehaviours[behaviourIndex].value > 0.01f)
226  {
227  if (blendingBehaviours[behaviourIndex].pose != 0)
228  {
229  // update additive animation only as needed
230  blendPoses[blendingBehaviours[behaviourIndex].pose].UpdateAdditiveAnimation(skeletonAction, inputSource);
231  }
232 
233  blendingBehaviours[behaviourIndex].ApplyBlending(snapshot, blendPoses, inputSource);
234  }
235  }

292  public void UpdateAdditiveAnimation(SteamVR_Action_Skeleton skeletonAction, SteamVR_Input_Sources inputSource)
293  {
294  SteamVR_Skeleton_PoseSnapshot snapshot = GetHandSnapshot(inputSource);
295  SteamVR_Skeleton_Pose_Hand poseHand = pose.GetHand(inputSource);
296 
297  for (int boneIndex = 0; boneIndex < snapshotL.bonePositions.Length; boneIndex++)
298  {
299  int fingerIndex = SteamVR_Skeleton_JointIndexes.GetFingerForBone(boneIndex);
300  SteamVR_Skeleton_FingerExtensionTypes extensionType = poseHand.GetMovementTypeForBone(boneIndex);
301 
302  if (extensionType == SteamVR_Skeleton_FingerExtensionTypes.Free)
303  {
304  snapshot.bonePositions[boneIndex] = skeletonAction.bonePositions[boneIndex];
305  snapshot.boneRotations[boneIndex] = skeletonAction.boneRotations[boneIndex];
306  }
307  if (extensionType == SteamVR_Skeleton_FingerExtensionTypes.Extend)
308  {
309  // lerp to open pose by fingercurl
310  snapshot.bonePositions[boneIndex] = Vector3.Lerp(poseHand.bonePositions[boneIndex], skeletonAction.bonePositions[boneIndex], 1 - skeletonAction.fingerCurls[fingerIndex]);
311  snapshot.boneRotations[boneIndex] = Quaternion.Lerp(poseHand.boneRotations[boneIndex], skeletonAction.boneRotations[boneIndex], 1 - skeletonAction.fingerCurls[fingerIndex]);
312  }
313  if (extensionType == SteamVR_Skeleton_FingerExtensionTypes.Contract)
314  {
315  // lerp to closed pose by fingercurl
316  snapshot.bonePositions[boneIndex] = Vector3.Lerp(poseHand.bonePositions[boneIndex], skeletonAction.bonePositions[boneIndex], skeletonAction.fingerCurls[fingerIndex]);
317  snapshot.boneRotations[boneIndex] = Quaternion.Lerp(poseHand.boneRotations[boneIndex], skeletonAction.boneRotations[boneIndex], skeletonAction.fingerCurls[fingerIndex]);
318  }
319  }
320  }

374  public void Update(float deltaTime, SteamVR_Input_Sources inputSource)
375  {
376  if (type == BlenderTypes.AnalogAction)
377  {
378  if (smoothingSpeed == 0)
379  value = action_single.GetAxis(inputSource);
380  else
381  value = Mathf.Lerp(value, action_single.GetAxis(inputSource), deltaTime * smoothingSpeed);
382  }
383  if (type == BlenderTypes.BooleanAction)
384  {
385  if (smoothingSpeed == 0)
386  value = action_bool.GetState(inputSource) ? 1 : 0;
387  else
388  value = Mathf.Lerp(value, action_bool.GetState(inputSource) ? 1 : 0, deltaTime * smoothingSpeed);
389  }
390  }

398  public void ApplyBlending(SteamVR_Skeleton_PoseSnapshot snapshot, SkeletonBlendablePose[] blendPoses, SteamVR_Input_Sources inputSource)
399  {
400  SteamVR_Skeleton_PoseSnapshot targetSnapshot = blendPoses[pose].GetHandSnapshot(inputSource);
401  if (mask.GetFinger(0) || useMask == false)
402  {
403  snapshot.position = Vector3.Lerp(snapshot.position, targetSnapshot.position, influence * value);
404  snapshot.rotation = Quaternion.Slerp(snapshot.rotation, targetSnapshot.rotation, influence * value);
405  }
406 
407  for (int boneIndex = 0; boneIndex < snapshot.bonePositions.Length; boneIndex++)
408  {
409  // verify the current finger is enabled in the mask, or if no mask is used.
410  if (mask.GetFinger(SteamVR_Skeleton_JointIndexes.GetFingerForBone(boneIndex) + 1) || useMask == false)
411  {
412  snapshot.bonePositions[boneIndex] = Vector3.Lerp(snapshot.bonePositions[boneIndex], targetSnapshot.bonePositions[boneIndex], influence * value);
413  snapshot.boneRotations[boneIndex] = Quaternion.Slerp(snapshot.boneRotations[boneIndex], targetSnapshot.boneRotations[boneIndex], influence * value);
414  }
415  }
416  }
