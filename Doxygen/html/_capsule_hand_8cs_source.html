237  private Dictionary<int, Mesh> _meshMap = new Dictionary<int, Mesh>();
238  private Mesh getCylinderMesh(float length) {
239  int lengthKey = Mathf.RoundToInt(length * 100 / CYLINDER_MESH_RESOLUTION);
240 
241  Mesh mesh;
242  if (_meshMap.TryGetValue(lengthKey, out mesh)) {
243  return mesh;
244  }
245 
246  mesh = new Mesh();
247  mesh.name = "GeneratedCylinder";
248  mesh.hideFlags = HideFlags.DontSave;
249 
250  List<Vector3> verts = new List<Vector3>();
251  List<Color> colors = new List<Color>();
252  List<int> tris = new List<int>();
253 
254  Vector3 p0 = Vector3.zero;
255  Vector3 p1 = Vector3.forward * length;
256  for (int i = 0; i < _cylinderResolution; i++) {
257  float angle = (Mathf.PI * 2.0f * i) / _cylinderResolution;
258  float dx = _cylinderRadius * Mathf.Cos(angle);
259  float dy = _cylinderRadius * Mathf.Sin(angle);
260 
261  Vector3 spoke = new Vector3(dx, dy, 0);
262 
263  verts.Add(p0 + spoke);
264  verts.Add(p1 + spoke);
265 
266  colors.Add(Color.white);
267  colors.Add(Color.white);
268 
269  int triStart = verts.Count;
270  int triCap = _cylinderResolution * 2;
271 
272  tris.Add((triStart + 0) % triCap);
273  tris.Add((triStart + 2) % triCap);
274  tris.Add((triStart + 1) % triCap);
275 
276  tris.Add((triStart + 2) % triCap);
277  tris.Add((triStart + 3) % triCap);
278  tris.Add((triStart + 1) % triCap);
279  }
280 
281  mesh.SetVertices(verts);
282  mesh.SetIndices(tris.ToArray(), MeshTopology.Triangles, 0);
283  mesh.RecalculateBounds();
284  mesh.RecalculateNormals();
285  mesh.UploadMeshData(true);
286 
287  _meshMap[lengthKey] = mesh;
288 
289  return mesh;
290  }
291  }
