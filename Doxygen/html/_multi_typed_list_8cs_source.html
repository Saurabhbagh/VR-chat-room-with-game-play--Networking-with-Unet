217  public override void RemoveAt(int index) {
218  var removedKey = _table[index];
219  _table.RemoveAt(index);
220 
221  getList(removedKey.id).RemoveAt(removedKey.index);
222 
223  for (int i = 0; i < _table.Count; i++) {
224  var key = _table[i];
225  if (key.id == removedKey.id && key.index > removedKey.index) {
226  key.index--;
227  _table[i] = key;
228  }
229  }
230  }

232  public override BaseType this[int index] {
233  get {
234  Key key = _table[index];
235  return (BaseType)getList(key.id)[key.index];
236  }
237  set {
238  Key oldKey = _table[index];
239  getList(oldKey.id).RemoveAt(oldKey.index);
240 
241  Key newKey = addInternal(value);
242  _table[index] = newKey;
243  }
244  }
