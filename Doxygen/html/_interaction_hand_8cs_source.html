348  protected override void unwarpColliders(Transform primaryHoverPoint,
349  ISpaceComponent warpedSpaceElement) {
350  // Extension method calculates "unwarped" pose in world space.
351  Vector3 unwarpedPosition;
352  Quaternion unwarpedRotation;
353  warpedSpaceElement.anchor.transformer.WorldSpaceUnwarp(primaryHoverPoint.position,
354  primaryHoverPoint.rotation,
355  out unwarpedPosition,
356  out unwarpedRotation);
357 
358  // First shift the hand to be centered on the fingertip position so that rotations
359  // applied to the hand will pivot around the fingertip, then apply the rest of the
360  // transformation.
361  _unwarpedHandData.Transform(-primaryHoverPoint.position, Quaternion.identity);
362  _unwarpedHandData.Transform(unwarpedPosition, unwarpedRotation
363  * Quaternion.Inverse(primaryHoverPoint.rotation));

424  private void initContactBones() {
425  _contactBones = new ContactBone[NUM_FINGERS * BONES_PER_FINGER + 1];
426  _handContactBoneMapFunctions = new BoneMapFunc[NUM_FINGERS * BONES_PER_FINGER + 1];
427 
428  // Finger bones
429  for (int fingerIndex = 0; fingerIndex < NUM_FINGERS; fingerIndex++) {
430  for (int jointIndex = 0; jointIndex < BONES_PER_FINGER; jointIndex++) {
431  GameObject contactBoneObj = new GameObject("Contact Fingerbone", typeof(CapsuleCollider), typeof(Rigidbody), typeof(ContactBone));
432  contactBoneObj.layer = manager.contactBoneLayer;
433 
434  Bone bone = _unwarpedHandData.Fingers[fingerIndex]
435  .Bone((Bone.BoneType)(jointIndex) + 1); // +1 to skip first bone.
436  int boneArrayIndex = fingerIndex * BONES_PER_FINGER + jointIndex;
437  contactBoneObj.transform.position = bone.Center.ToVector3();
438  contactBoneObj.transform.rotation = bone.Rotation.ToQuaternion();
439 
440  // Remember the method we used to calculate this bone position from
441  // a Leap Hand for later.
442  int fingerIndexCopy = fingerIndex;
443  int jointIndexCopy = jointIndex;
444  _handContactBoneMapFunctions[boneArrayIndex] = (Leap.Hand hand,
445  out Vector3 targetPosition,
446  out Quaternion targetRotation) => {
447  Bone theBone = hand.Fingers[fingerIndexCopy].Bone((Bone.BoneType)(jointIndexCopy + 1));
448  targetPosition = theBone.Center.ToVector3();
449  targetRotation = theBone.Rotation.ToQuaternion();
450  };
451 
452  CapsuleCollider capsule = contactBoneObj.GetComponent<CapsuleCollider>();
453  capsule.direction = 2;
454  capsule.radius = bone.Width * 0.5f;
455  capsule.height = bone.Length + bone.Width;
456  capsule.material = defaultContactBoneMaterial;
457 
458  ContactBone contactBone = initContactBone(bone, contactBoneObj, boneArrayIndex, capsule);
459 
460  contactBone.lastTargetPosition = bone.Center.ToVector3();
461  }
462  }
463 
464  // Palm bone
465  {
466  // Palm is attached to the third metacarpal and derived from it.
467  GameObject contactBoneObj = new GameObject("Contact Palm Bone", typeof(BoxCollider), typeof(Rigidbody), typeof(ContactBone));
468 
469  Bone bone = _unwarpedHandData.Fingers[(int)Finger.FingerType.TYPE_MIDDLE].Bone(Bone.BoneType.TYPE_METACARPAL);
470  int boneArrayIndex = NUM_FINGERS * BONES_PER_FINGER;
471  contactBoneObj.transform.position = _unwarpedHandData.PalmPosition.ToVector3();
472  contactBoneObj.transform.rotation = _unwarpedHandData.Rotation.ToQuaternion();
473 
474  // Remember the method we used to calculate the palm from a Leap Hand for later.
475  _handContactBoneMapFunctions[boneArrayIndex] = (Leap.Hand hand,
476  out Vector3 targetPosition,
477  out Quaternion targetRotation) => {
478  targetPosition = hand.PalmPosition.ToVector3();
479  targetRotation = hand.Rotation.ToQuaternion();
480  };
481 
482  BoxCollider box = contactBoneObj.GetComponent<BoxCollider>();
483  box.center = new Vector3(_unwarpedHandData.IsLeft ? -0.005f : 0.005f, bone.Width * -0.3f, -0.01f);
484  box.size = new Vector3(bone.Length, bone.Width, bone.Length);

524  private void addContactBoneJoints() {
525  for (int fingerIndex = 0; fingerIndex < NUM_FINGERS; fingerIndex++) {
526  for (int jointIndex = 0; jointIndex < BONES_PER_FINGER; jointIndex++) {
527  Bone bone = _unwarpedHandData.Fingers[fingerIndex].Bone((Bone.BoneType)(jointIndex) + 1); // +1 to skip first bone.
528  int boneArrayIndex = fingerIndex * BONES_PER_FINGER + jointIndex;
529 
530  FixedJoint joint = _contactBones[boneArrayIndex].gameObject.AddComponent<FixedJoint>();
531  joint.autoConfigureConnectedAnchor = false;
532  if (jointIndex != 0) {
533  Bone prevBone = _unwarpedHandData.Fingers[fingerIndex].Bone((Bone.BoneType)(jointIndex));
534  joint.connectedBody = _contactBones[boneArrayIndex - 1].rigidbody;
535  joint.anchor = Vector3.back * bone.Length / 2f;
536  joint.connectedAnchor = Vector3.forward * prevBone.Length / 2f;
537  _contactBones[boneArrayIndex].joint = joint;
538  }
539  else {
540  joint.connectedBody = _contactBones[NUM_FINGERS * BONES_PER_FINGER].rigidbody;
541  joint.anchor = Vector3.back * bone.Length / 2f;
542  joint.connectedAnchor = _contactBones[NUM_FINGERS * BONES_PER_FINGER].transform.InverseTransformPoint(bone.PrevJoint.ToVector3());
543  _contactBones[boneArrayIndex].metacarpalJoint = joint;
544  }
545  }
546  }

550  private void resetContactBoneJoints() {
551  // If contact bones array is null, there's nothing to reset. This can happen if
552  // the controller is disabled before it had a chance to initialize contact.
553  if (_contactBones == null) return;
554 
555  // If the palm contact bone is null, we can't reset bone joints.
556  if (_contactBones[NUM_FINGERS * BONES_PER_FINGER] == null) return;
557 
558  _contactBones[NUM_FINGERS * BONES_PER_FINGER].transform.position = _unwarpedHandData.PalmPosition.ToVector3();
559  _contactBones[NUM_FINGERS * BONES_PER_FINGER].transform.rotation = _unwarpedHandData.Rotation.ToQuaternion();
560  _contactBones[NUM_FINGERS * BONES_PER_FINGER].rigidbody.velocity = Vector3.zero;
561  _contactBones[NUM_FINGERS * BONES_PER_FINGER].rigidbody.angularVelocity = Vector3.zero;
562 
563  for (int fingerIndex = 0; fingerIndex < NUM_FINGERS; fingerIndex++) {
564  for (int jointIndex = 0; jointIndex < BONES_PER_FINGER; jointIndex++) {
565  Bone bone = _unwarpedHandData.Fingers[fingerIndex].Bone((Bone.BoneType)(jointIndex) + 1); // +1 to skip first bone.
566  int boneArrayIndex = fingerIndex * BONES_PER_FINGER + jointIndex;
567 
568  _contactBones[boneArrayIndex].transform.position = bone.Center.ToVector3();
569  _contactBones[boneArrayIndex].transform.rotation = bone.Rotation.ToQuaternion();
570  _contactBones[boneArrayIndex].rigidbody.position = bone.Center.ToVector3();
571  _contactBones[boneArrayIndex].rigidbody.rotation = bone.Rotation.ToQuaternion();
572  _contactBones[boneArrayIndex].rigidbody.velocity = Vector3.zero;
573  _contactBones[boneArrayIndex].rigidbody.angularVelocity = Vector3.zero;
574 
575  if (jointIndex != 0 && _contactBones[boneArrayIndex].joint != null) {
576  Bone prevBone = _unwarpedHandData.Fingers[fingerIndex].Bone((Bone.BoneType)(jointIndex));
577  _contactBones[boneArrayIndex].joint.connectedBody = _contactBones[boneArrayIndex - 1].rigidbody;
578  _contactBones[boneArrayIndex].joint.anchor = Vector3.back * bone.Length / 2f;
579  _contactBones[boneArrayIndex].joint.connectedAnchor = Vector3.forward * prevBone.Length / 2f;
580  }
581  else if (_contactBones[boneArrayIndex].metacarpalJoint != null) {
582  _contactBones[boneArrayIndex].metacarpalJoint.connectedBody = _contactBones[NUM_FINGERS * BONES_PER_FINGER].rigidbody;
583  _contactBones[boneArrayIndex].metacarpalJoint.anchor = Vector3.back * bone.Length / 2f;
584  _contactBones[boneArrayIndex].metacarpalJoint.connectedAnchor = _contactBones[NUM_FINGERS * BONES_PER_FINGER].transform
585  .InverseTransformPoint(bone.PrevJoint.ToVector3());
586  }
587  }
588  }

595  public void FillBones(Hand inHand) {
596  if (softContactEnabled) { return; }
597  if (Application.isPlaying && _contactBones.Length == NUM_FINGERS * BONES_PER_FINGER + 1) {
598  Vector elbowPos = inHand.Arm.ElbowPosition;
599  inHand.SetTransform(_contactBones[NUM_FINGERS * BONES_PER_FINGER].rigidbody.position, _contactBones[NUM_FINGERS * BONES_PER_FINGER].rigidbody.rotation);
600 
601  for (int fingerIndex = 0; fingerIndex < NUM_FINGERS; fingerIndex++) {
602  for (int jointIndex = 0; jointIndex < BONES_PER_FINGER; jointIndex++) {
603  Bone bone = inHand.Fingers[fingerIndex].Bone((Bone.BoneType)(jointIndex) + 1);
604  int boneArrayIndex = fingerIndex * BONES_PER_FINGER + jointIndex;
605  Vector displacement = _contactBones[boneArrayIndex].rigidbody.position.ToVector() - bone.Center;
606  bone.Center += displacement;
607  bone.PrevJoint += displacement;
608  bone.NextJoint += displacement;
609  bone.Rotation = _contactBones[boneArrayIndex].rigidbody.rotation.ToLeapQuaternion();
610  }
611  }
612 
613  inHand.Arm.PrevJoint = elbowPos;
614  inHand.Arm.Direction = (inHand.Arm.PrevJoint - inHand.Arm.NextJoint).Normalized;
615  inHand.Arm.Center = (inHand.Arm.PrevJoint + inHand.Arm.NextJoint) * 0.5f;
616  }
617  }

663  private Vector3[] _fingertipPositionsBuffer = new Vector3[5];
671  public override Vector3 GetGraspPoint() {
672  if (!isGraspingObject) {
673  Debug.LogError("Tried to get grasp point of InteractionHand, but it is not "
674  + "currently grasping an object.", this);
675  return leapHand.PalmPosition.ToVector3();
676  }
677 
678  int numGraspingFingertips = 0;
679  _grabClassifier.GetGraspingFingertipPositions(graspedObject, _fingertipPositionsBuffer, out numGraspingFingertips);
680  if (numGraspingFingertips > 0) {
681  Vector3 sum = Vector3.zero;
682  for (int i = 0; i < numGraspingFingertips; i++) {
683  sum += _fingertipPositionsBuffer[i];
