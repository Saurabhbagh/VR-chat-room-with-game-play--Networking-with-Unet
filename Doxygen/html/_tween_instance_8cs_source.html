102  public void interpolatePercent() {
103  float progress;
104  switch (smoothType) {
105  case SmoothType.Linear:
106  progress = curPercent;
107  break;
108  case SmoothType.Smooth:
109  progress = Mathf.SmoothStep(0, 1, curPercent);
110  break;
111  case SmoothType.SmoothEnd:
112  progress = 1.0f - (curPercent - 1.0f) * (curPercent - 1.0f);
113  break;
114  case SmoothType.SmoothStart:
115  progress = curPercent * curPercent;
116  break;
117  default:
118  progress = smoothFunction(curPercent);
119  break;
120  }
