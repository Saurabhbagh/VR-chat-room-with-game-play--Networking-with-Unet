169  IEnumerator SetModelAsync(string newRenderModelName)
170  {
171  meshRenderers.Clear();
172 
173  if (string.IsNullOrEmpty(newRenderModelName))
174  yield break;
175 
176  // Preload all render models before asking for the data to create meshes.
177  using (RenderModelInterfaceHolder holder = new RenderModelInterfaceHolder())
178  {
179  CVRRenderModels renderModels = holder.instance;
180  if (renderModels == null)
181  yield break;
182 
183  // Gather names of render models to preload.
184  string[] renderModelNames;
185 
186  uint count = renderModels.GetComponentCount(newRenderModelName);
187  if (count > 0)
188  {
189  renderModelNames = new string[count];
190 
191  for (int componentIndex = 0; componentIndex < count; componentIndex++)
192  {
193  uint capacity = renderModels.GetComponentName(newRenderModelName, (uint)componentIndex, null, 0);
194  if (capacity == 0)
195  continue;
196 
197  var componentNameStringBuilder = new System.Text.StringBuilder((int)capacity);
198  if (renderModels.GetComponentName(newRenderModelName, (uint)componentIndex, componentNameStringBuilder, capacity) == 0)
199  continue;
200 
201  string componentName = componentNameStringBuilder.ToString();
202 
203  capacity = renderModels.GetComponentRenderModelName(newRenderModelName, componentName, null, 0);
204  if (capacity == 0)
205  continue;
206 
207  var nameStringBuilder = new System.Text.StringBuilder((int)capacity);
208  if (renderModels.GetComponentRenderModelName(newRenderModelName, componentName, nameStringBuilder, capacity) == 0)
209  continue;
210 
211  var s = nameStringBuilder.ToString();
212 
213  // Only need to preload if not already cached.
214  RenderModel model = models[s] as RenderModel;
215  if (model == null || model.mesh == null)
216  {
217  renderModelNames[componentIndex] = s;
218  }
219  }
220  }
221  else
222  {
223  // Only need to preload if not already cached.
224  RenderModel model = models[newRenderModelName] as RenderModel;
225  if (model == null || model.mesh == null)
226  {
227  renderModelNames = new string[] { newRenderModelName };
228  }
229  else
230  {
231  renderModelNames = new string[0];
232  }
233  }
234 
235  // Keep trying every 100ms until all components finish loading.
236  while (true)
237  {
238  var loading = false;
239  for (int renderModelNameIndex = 0; renderModelNameIndex < renderModelNames.Length; renderModelNameIndex++)
240  {
241  if (string.IsNullOrEmpty(renderModelNames[renderModelNameIndex]))
242  continue;
243 
244  var pRenderModel = System.IntPtr.Zero;
245 
246  var error = renderModels.LoadRenderModel_Async(renderModelNames[renderModelNameIndex], ref pRenderModel);
247  //Debug.Log("<b>[SteamVR]</b> renderModels.LoadRenderModel_Async(" + renderModelNames[renderModelNameIndex] + ": " + error.ToString());
248 
249  if (error == EVRRenderModelError.Loading)
250  {
251  loading = true;
252  }
253  else if (error == EVRRenderModelError.None)
254  {
255  // Preload textures as well.
256  var renderModel = MarshalRenderModel(pRenderModel);
257 
258  // Check the cache first.
259  var material = materials[renderModel.diffuseTextureId] as Material;
260  if (material == null || material.mainTexture == null)
261  {
262  var pDiffuseTexture = System.IntPtr.Zero;
263 
264  error = renderModels.LoadTexture_Async(renderModel.diffuseTextureId, ref pDiffuseTexture);
265  //Debug.Log("<b>[SteamVR]</b> renderModels.LoadRenderModel_Async(" + renderModelNames[renderModelNameIndex] + ": " + error.ToString());
266 
267  if (error == EVRRenderModelError.Loading)
268  {
269  loading = true;
270  }
271  }
272  }
273  }
274 
275  if (loading)
276  {
277  yield return new WaitForSecondsRealtime(0.1f);
278  }
279  else
280  {
281  break;
282  }
283  }
284  }
285 
286  bool success = SetModel(newRenderModelName);
287  this.renderModelName = newRenderModelName;
288  SteamVR_Events.RenderModelLoaded.Send(this, success);
289  }

340  RenderModel LoadRenderModel(CVRRenderModels renderModels, string renderModelName, string baseName)
341  {
342  var pRenderModel = System.IntPtr.Zero;
343 
344  EVRRenderModelError error;
345  while (true)
346  {
347  error = renderModels.LoadRenderModel_Async(renderModelName, ref pRenderModel);
348  if (error != EVRRenderModelError.Loading)
349  break;
350 
351  Sleep();
352  }
353 
354  if (error != EVRRenderModelError.None)
355  {
356  Debug.LogError(string.Format("<b>[SteamVR]</b> Failed to load render model {0} - {1}", renderModelName, error.ToString()));
357  return null;
358  }
359 
360  var renderModel = MarshalRenderModel(pRenderModel);
361 
362  var vertices = new Vector3[renderModel.unVertexCount];
363  var normals = new Vector3[renderModel.unVertexCount];
364  var uv = new Vector2[renderModel.unVertexCount];
365 
366  var type = typeof(RenderModel_Vertex_t);
367  for (int iVert = 0; iVert < renderModel.unVertexCount; iVert++)
368  {
369  var ptr = new System.IntPtr(renderModel.rVertexData.ToInt64() + iVert * Marshal.SizeOf(type));
370  var vert = (RenderModel_Vertex_t)Marshal.PtrToStructure(ptr, type);
371 
372  vertices[iVert] = new Vector3(vert.vPosition.v0, vert.vPosition.v1, -vert.vPosition.v2);
373  normals[iVert] = new Vector3(vert.vNormal.v0, vert.vNormal.v1, -vert.vNormal.v2);
374  uv[iVert] = new Vector2(vert.rfTextureCoord0, vert.rfTextureCoord1);
375  }
376 
377  int indexCount = (int)renderModel.unTriangleCount * 3;
378  var indices = new short[indexCount];
379  Marshal.Copy(renderModel.rIndexData, indices, 0, indices.Length);
380 
381  var triangles = new int[indexCount];
382  for (int iTri = 0; iTri < renderModel.unTriangleCount; iTri++)
383  {
384  triangles[iTri * 3 + 0] = (int)indices[iTri * 3 + 2];
385  triangles[iTri * 3 + 1] = (int)indices[iTri * 3 + 1];
386  triangles[iTri * 3 + 2] = (int)indices[iTri * 3 + 0];
387  }
388 
389  var mesh = new Mesh();
390  mesh.vertices = vertices;
391  mesh.normals = normals;
392  mesh.uv = uv;
393  mesh.triangles = triangles;
394 
395 #if (UNITY_5_4 || UNITY_5_3 || UNITY_5_2 || UNITY_5_1 || UNITY_5_0)
396  mesh.Optimize();
397 #endif
398  //mesh.hideFlags = HideFlags.DontUnloadUnusedAsset;
399 
400  // Check cache before loading texture.
401  var material = materials[renderModel.diffuseTextureId] as Material;
402  if (material == null || material.mainTexture == null)
403  {
404  var pDiffuseTexture = System.IntPtr.Zero;
405 
406  while (true)
407  {
408  error = renderModels.LoadTexture_Async(renderModel.diffuseTextureId, ref pDiffuseTexture);
409  if (error != EVRRenderModelError.Loading)
410  break;
411 
412  Sleep();
413  }
414 
415  if (error == EVRRenderModelError.None)
416  {
417  var diffuseTexture = MarshalRenderModel_TextureMap(pDiffuseTexture);
418  var texture = new Texture2D(diffuseTexture.unWidth, diffuseTexture.unHeight, TextureFormat.RGBA32, false);
419  if (SystemInfo.graphicsDeviceType == UnityEngine.Rendering.GraphicsDeviceType.Direct3D11)
420  {
421  texture.Apply();
422  System.IntPtr texturePointer = texture.GetNativeTexturePtr();
423  while (true)
424  {
425  error = renderModels.LoadIntoTextureD3D11_Async(renderModel.diffuseTextureId, texturePointer);
426  if (error != EVRRenderModelError.Loading)
427  break;
428 
429  Sleep();
430  }
431  }
432  else
433  {
434  var textureMapData = new byte[diffuseTexture.unWidth * diffuseTexture.unHeight * 4]; // RGBA
435  Marshal.Copy(diffuseTexture.rubTextureMapData, textureMapData, 0, textureMapData.Length);
436 
437  var colors = new Color32[diffuseTexture.unWidth * diffuseTexture.unHeight];
438  int iColor = 0;
439  for (int iHeight = 0; iHeight < diffuseTexture.unHeight; iHeight++)
440  {
441  for (int iWidth = 0; iWidth < diffuseTexture.unWidth; iWidth++)
442  {
443  var r = textureMapData[iColor++];
444  var g = textureMapData[iColor++];
445  var b = textureMapData[iColor++];
446  var a = textureMapData[iColor++];
447  colors[iHeight * diffuseTexture.unWidth + iWidth] = new Color32(r, g, b, a);
448  }
449  }
450 
451  texture.SetPixels32(colors);
452  texture.Apply();
453  }
454 
455  material = new Material(shader != null ? shader : Shader.Find("Standard"));
456  material.mainTexture = texture;
457  //material.hideFlags = HideFlags.DontUnloadUnusedAsset;
458 
459  materials[renderModel.diffuseTextureId] = material;
460 
461  renderModels.FreeTexture(pDiffuseTexture);
462  }
463  else
464  {
465  Debug.Log("<b>[SteamVR]</b> Failed to load render model texture for render model " + renderModelName + ". Error: " + error.ToString());
466  }
467  }
468 
469  // Delay freeing when we can since we'll often get multiple requests for the same model right
470  // after another (e.g. two controllers or two basestations).
471 #if UNITY_EDITOR
472  if (!Application.isPlaying)
473  renderModels.FreeRenderModel(pRenderModel);
474  else
475 #endif
476  StartCoroutine(FreeRenderModel(pRenderModel));
477 
478  return new RenderModel(mesh, material);
479  }

529  private bool LoadComponents(RenderModelInterfaceHolder holder, string renderModelName)
530  {
531  // Disable existing components (we will re-enable them if referenced by this new model).
532  // Also strip mesh filter and renderer since these will get re-added if the new component needs them.
533  var t = transform;
534  for (int childIndex = 0; childIndex < t.childCount; childIndex++)
535  {
536  var child = t.GetChild(childIndex);
537  child.gameObject.SetActive(false);
538  StripMesh(child.gameObject);
539  }
540 
541  // If no model specified, we're done; return success.
542  if (string.IsNullOrEmpty(renderModelName))
543  return true;
544 
545  var renderModels = holder.instance;
546  if (renderModels == null)
547  return false;
548 
549  var count = renderModels.GetComponentCount(renderModelName);
550  if (count == 0)
551  return false;
552 
553  for (int i = 0; i < count; i++)
554  {
555  var capacity = renderModels.GetComponentName(renderModelName, (uint)i, null, 0);
556  if (capacity == 0)
557  continue;
558 
559  System.Text.StringBuilder componentNameStringBuilder = new System.Text.StringBuilder((int)capacity);
560  if (renderModels.GetComponentName(renderModelName, (uint)i, componentNameStringBuilder, capacity) == 0)
561  continue;
562 
563  string componentName = componentNameStringBuilder.ToString();
564 
565  // Create (or reuse) a child object for this component (some components are dynamic and don't have meshes).
566  t = FindTransformByName(componentName);
567  if (t != null)
568  {
569  t.gameObject.SetActive(true);
570  componentAttachPoints[componentName] = FindTransformByName(k_localTransformName, t);
571  }
572  else
573  {
574  t = new GameObject(componentName).transform;
575  t.parent = transform;
576  t.gameObject.layer = gameObject.layer;
577 
578  // Also create a child 'attach' object for attaching things.
579  var attach = new GameObject(k_localTransformName).transform;
580  attach.parent = t;
581  attach.localPosition = Vector3.zero;
582  attach.localRotation = Quaternion.identity;
583  attach.localScale = Vector3.one;
584  attach.gameObject.layer = gameObject.layer;
585 
586  componentAttachPoints[componentName] = attach;
587  }
588 
589  // Reset transform.
590  t.localPosition = Vector3.zero;
591  t.localRotation = Quaternion.identity;
592  t.localScale = Vector3.one;
593 
594  capacity = renderModels.GetComponentRenderModelName(renderModelName, componentName, null, 0);
595  if (capacity == 0)
596  continue;
597 
598  var componentRenderModelNameStringBuilder = new System.Text.StringBuilder((int)capacity);
599  if (renderModels.GetComponentRenderModelName(renderModelName, componentName, componentRenderModelNameStringBuilder, capacity) == 0)
600  continue;
601 
602  string componentRenderModelName = componentRenderModelNameStringBuilder.ToString();
603 
604  // Check the cache or load into memory.
605  var model = models[componentRenderModelName] as RenderModel;
606  if (model == null || model.mesh == null)
607  {
608  if (verbose)
609  Debug.Log("<b>[SteamVR]</b> Loading render model " + componentRenderModelName);
610 
611  model = LoadRenderModel(renderModels, componentRenderModelName, renderModelName);
612  if (model == null)
613  continue;
614 
615  models[componentRenderModelName] = model;
616  }
617 
618  t.gameObject.AddComponent<MeshFilter>().mesh = model.mesh;
619  MeshRenderer newRenderer = t.gameObject.AddComponent<MeshRenderer>();
620  newRenderer.sharedMaterial = model.material;
621  meshRenderers.Add(newRenderer);
622  }
623 
624  return true;
625  }

674  void Update()
675  {
676 #if UNITY_EDITOR
677  if (!Application.isPlaying)
678  {
679  // See if anything has changed since this gets called whenever anything gets touched.
680  var fields = GetType().GetFields(System.Reflection.BindingFlags.Instance | System.Reflection.BindingFlags.Public);
681 
682  bool modified = false;
683 
684  if (values == null)
685  {
686  modified = true;
687  }
688  else
689  {
690  foreach (var f in fields)
691  {
692  if (!values.Contains(f))
693  {
694  modified = true;
695  break;
696  }
697 
698  var v0 = values[f];
699  var v1 = f.GetValue(this);
700  if (v1 != null)
701  {
702  if (!v1.Equals(v0))
703  {
704  modified = true;
705  break;
706  }
707  }
708  else if (v0 != null)
709  {
710  modified = true;
711  break;
712  }
713  }
714  }
715 
716  if (modified)
717  {
718  if (renderModelName != modelOverride)
719  {
720  renderModelName = modelOverride;
721  SetModel(modelOverride);
722  }
723 
724  values = new Hashtable();
725  foreach (var f in fields)
726  values[f] = f.GetValue(this);
727  }
728 
729  return; // Do not update transforms (below) when not playing in Editor (to avoid keeping OpenVR running all the time).
730  }
731 #endif
732  // Update component transforms dynamically.
733  if (updateDynamically)
734  UpdateComponents(OpenVR.RenderModels);
735  }

740  {
741  if (renderModels == null)
742  return;
