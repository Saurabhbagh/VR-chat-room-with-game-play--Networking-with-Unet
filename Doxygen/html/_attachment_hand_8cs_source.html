187  public void refreshAttachmentTransforms(AttachmentPointFlags points) {
188  if (_attachmentPointFlagConstants == null || _attachmentPointFlagConstants.Length == 0) {
189  initializeAttachmentPointFlagConstants();
190  }
191 
192  // First just _check_ whether we'll need to do any destruction or creation
193  bool requiresDestructionOrCreation = false;
194  foreach (var flag in _attachmentPointFlagConstants) {
195  if (flag == AttachmentPointFlags.None) continue;
196 
197  if ((!points.Contains(flag) && GetBehaviourForPoint(flag) != null)
198  || (points.Contains(flag) && GetBehaviourForPoint(flag) == null)) {
199  requiresDestructionOrCreation = true;
200  }
201  }
202 
203  // Go through the work of flattening and rebuilding if it is necessary.
204  if (requiresDestructionOrCreation) {
205  // Remove parent-child relationships so deleting parent Transforms doesn't annihilate
206  // child Transforms that don't need to be deleted themselves.
207  flattenAttachmentTransformHierarchy();
208 
209  foreach (AttachmentPointFlags flag in _attachmentPointFlagConstants) {
210  if (flag == AttachmentPointFlags.None) continue;
211 
212  if (points.Contains(flag)) {
213  ensureTransformExists(flag);
214  }
215  else {
216  ensureTransformDoesNotExist(flag);
217  }
218  }
219 
220  // Organize transforms, restoring parent-child relationships.
221  organizeAttachmentTransforms();
222  }
223 
224  if (_attachmentPointsDirty) {
225  OnAttachmentPointsModified();
226  _attachmentPointsDirty = false;
227  }
228  }

295  private void ensureTransformExists(AttachmentPointFlags singlePoint) {
296  if (!singlePoint.IsSinglePoint()) {
297  Debug.LogError("Tried to ensure transform exists for singlePoint, but it contains more than one set flag.");
298  return;
299  }
300 
301  AttachmentPointBehaviour pointBehaviour = GetBehaviourForPoint(singlePoint);
302 
303  if (pointBehaviour == null) {
304  // First, see if there's already one in the hierarchy! Might exist due to, e.g. an Undo operation
305  var existingPointBehaviour = this.gameObject.GetComponentsInChildren<AttachmentPointBehaviour>()
306  .Query()
307  .FirstOrDefault(p => p.attachmentPoint == singlePoint);
308 
309  // Only make a new object if the transform really doesn't exist.
310  if (existingPointBehaviour == AttachmentPointFlags.None) {
311  GameObject obj = new GameObject(Enum.GetName(typeof(AttachmentPointFlags), singlePoint));
312  #if UNITY_EDITOR
313  Undo.RegisterCreatedObjectUndo(obj, "Created Object");
314  pointBehaviour = Undo.AddComponent<AttachmentPointBehaviour>(obj);
315  #else
316  pointBehaviour = obj.AddComponent<AttachmentPointBehaviour>();
317  #endif
318  }
319  else {
320  pointBehaviour = existingPointBehaviour;
321  }
322 
323  #if UNITY_EDITOR
324  Undo.RecordObject(pointBehaviour, "Set Attachment Point");
325  #endif
326  pointBehaviour.attachmentPoint = singlePoint;
327  pointBehaviour.attachmentHand = this;
328  setBehaviourForPoint(singlePoint, pointBehaviour);
329 
330  SetTransformParent(pointBehaviour.transform, this.transform);
331 
332  _attachmentPointsDirty = true;
333 
334  #if UNITY_EDITOR
335  EditorUtility.SetDirty(this);
336  #endif
337  }

435  private static Transform[] s_hierarchyTransformsBuffer = new Transform[4];
441  private Transform tryStackTransformHierarchy(params Transform[] transforms) {
442  for (int i = 0; i < s_hierarchyTransformsBuffer.Length; i++) {
443  s_hierarchyTransformsBuffer[i] = null;
444  }
445 
446  int hierarchyCount = 0;
447 
448  foreach (var transform in transforms.Query().Where(t => t != null)) {
449  s_hierarchyTransformsBuffer[hierarchyCount++] = transform;
450  }
451 
452  for (int i = hierarchyCount - 1; i > 0; i--) {
453  SetTransformParent(s_hierarchyTransformsBuffer[i], s_hierarchyTransformsBuffer[i - 1]);
454  }
455 
456  if (hierarchyCount > 0) {
457  return s_hierarchyTransformsBuffer[0];
458  }
459 
460  return null;
461  }

463  private static Transform[] s_transformsBuffer = new Transform[4];
464  private Transform tryStackTransformHierarchy(params MonoBehaviour[] monoBehaviours) {
465  for (int i = 0; i < s_transformsBuffer.Length; i++) {
466  s_transformsBuffer[i] = null;
467  }
468 
469  int tIdx = 0;
470  foreach (var behaviour in monoBehaviours.Query().Where(b => b != null)) {
471  s_transformsBuffer[tIdx++] = behaviour.transform;
472  }
473 
474  return tryStackTransformHierarchy(s_transformsBuffer);
475  }

481  public struct AttachmentPointsEnumerator {

484  private int _flagsCount;

486  public AttachmentPointsEnumerator GetEnumerator() { return this; }
487 
488  public AttachmentPointsEnumerator(AttachmentHand hand) {
489  if (hand != null && hand._attachmentPointFlagConstants != null) {
490  _curIdx = -1;
491  _hand = hand;
492  _flagsCount = hand._attachmentPointFlagConstants.Length;
493  }
494  else {
495  // Hand doesn't exist (destroyed?) or isn't initialized yet.
496  _curIdx = -1;
497  _hand = null;
498  _flagsCount = 0;
499  }
500  }
501 
502  public AttachmentPointBehaviour Current {
503  get {
504  if (_hand == null) return null;
505 
506  return _hand.GetBehaviourForPoint(GetFlagFromFlagIdx(_curIdx));
507  }
508  }
509 
510  public bool MoveNext() {
511  do {
512  _curIdx++;
513  } while (_curIdx < _flagsCount && _hand.GetBehaviourForPoint(GetFlagFromFlagIdx(_curIdx)) == null);
514 
515  return _curIdx < _flagsCount;
516  }
517  }

519  private static AttachmentPointFlags GetFlagFromFlagIdx(int pointIdx) {
