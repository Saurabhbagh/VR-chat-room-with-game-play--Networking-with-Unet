306  protected virtual void FixedUpdate() {
307  if (_frameOptimization == FrameOptimizationMode.ReuseUpdateForPhysics) {
308  DispatchFixedFrameEvent(_transformedUpdateFrame);
309  return;
310  }
311 
312  if (_useInterpolation) {
313 
314  long timestamp;
315  switch (_frameOptimization) {
316  case FrameOptimizationMode.None:
317  // By default we use Time.fixedTime to ensure that our hands are on the same
318  // timeline as Update. We add an extrapolation value to help compensate
319  // for latency.
320  float extrapolatedTime = Time.fixedTime + CalculatePhysicsExtrapolation();
321  timestamp = (long)(extrapolatedTime * S_TO_NS) + _unityToLeapOffset;
322  break;
323  case FrameOptimizationMode.ReusePhysicsForUpdate:
324  // If we are re-using physics frames for update, we don't even want to care
325  // about Time.fixedTime, just grab the most recent interpolated timestamp
326  // like we are in Update.
327  timestamp = CalculateInterpolationTime() + (ExtrapolationAmount * 1000);
328  break;
329  default:
330  throw new System.InvalidOperationException(
331  "Unexpected frame optimization mode: " + _frameOptimization);
332  }
333  _leapController.GetInterpolatedFrame(_untransformedFixedFrame, timestamp);
334 
335  }
336  else {
337  _leapController.Frame(_untransformedFixedFrame);
338  }
339 
340  if (_untransformedFixedFrame != null) {
341  transformFrame(_untransformedFixedFrame, _transformedFixedFrame);
342 
343  DispatchFixedFrameEvent(_transformedFixedFrame);
344  }
345  }
