691  public static bool DetectFor(Transform transform, OldRigHierarchy toFill) {
692  if (transform == null) return false;
693 
694  var didDetectOldRig = false;
695  var scannedRig = toFill;
696  scannedRig.rigTransform = transform;
697 
698  // Scan for the camera transform in immediate children.
699  var camDataBuffer = new List<OldCameraData>();
700  camDataBuffer.Clear();
701  detectPotentialOldRigCameras(transform, camDataBuffer);
702 
703  if (camDataBuffer.Count == 0) {
704  // No cameras found; this is not an upgradeable old rig.
705  return false;
706  }
707  else {
708  // We have a list of children that contain Camera components and potentially
709  // other Leap rig-related data, but the check was intentionally loose.
710 
711  // Some practical cases to consider, if there was only one camera found:
712  // (1) It could be a standard non-image rig, and so is upgradeable!
713  // (2) It could be an image rig but with one eye removed (weird).
714  // (2a) If the left eye was removed, the rig is non-upgradeable.
715  // (2b) If the right eye was removed, this rig will look identical to a
716  // non-image rig as in (1) and be upgradeable.
717  // (3) It could just be an object with a Camera component beneath it, and be
718  // unrelated to a Leap VR rig. (Looks similar to 2a but with fewer
719  // Leap-related or Missing components.)
720  //
721  // To confirm it is NOT (3) or (2a), we need to check its children to
722  // find the LeapSpace and LeapHandController.
723 
724  // Alternatively, if there were two potential cameras found:
725  // (1) Optimistically, this could indicate it's an old image rig with a left
726  // and right eye.
727  // (2) It could be a stranger construction, like the rig for a custom VR
728  // integration.
729  // (3) It could be a standard single-camera non-image rig with another Camera
730  // utilized for some other, distinct purpose.
731  // (4) It could just be a transform with two cameras as children and not be
732  // Leap-related!
733 
734  // To be an upgradeable Leap Rig, image or otherwise, one of the cameras we've
735  // found must have a LeapServiceProvider two transforms down (we don't actually
736  // care what transforms are named, since someone could change them without
737  // really affecting the structure of the rig):
738  //
739  // (Camera)
740  // |
741  // `- LeapSpace - Missing Component // optional, formerly LeapVRTemporalWarping
742  // |
743  // `- LeapHandController - LeapServiceProvider // required
744  // HandModelManager // optional
745  //
746  // If there is no LeapServiceProvider two-transforms-down, the rig has been
747  // modified too heavily and is no longer auto-upgradeable.
748  //
749  // Importantly, this condition will hold true if the rig is a non-image rig OR
750  // an image rig, and false if we are going to refuse the upgrade. What's more,
751  // if this condition holds true, we will be able to collect all the information
752  // we need from the rig after scanning through each camera.
753 
754  var primaryCamData = (OldCameraData)null;
755  var firstLeapSpaceData = (OldLeapSpaceData)null;
756  var firstLHCData = (OldLeapHandControllerData)null;
757 
758  // "Secondary" cameras in the rig are the cameras that don't have LeapSpace or
759  // LeapHandController objects (or weren't searched for these if we already
760  // found a primary camera).
761  // These need to be searched after we find the primary camera to find the
762  // right -eye camera in the image rig case.
763  var secondaryCamDatas = new List<OldCameraData>();
764 
765  for (int i = 0; i < camDataBuffer.Count; i++) {
766  var camData = camDataBuffer[i];
767  var expectingFirstLeapSpaceData = firstLeapSpaceData == null;
768 
769  var camTransform = camData.cameraTransform;
770  foreach (var child in camTransform.GetChildren()) {
771  var expectingFirstLHCData = firstLHCData == null;
772 
773  // We're looking for the child that is the LeapSpace transform, only
774  // known implicitly by looking at _its_ children.
775  foreach (var grandchild in child.GetChildren()) {
776  var leapServiceProvider = grandchild.GetComponent<LeapServiceProvider>();
777 
778  // We're going to assume the first LeapServiceProvider we find is
779  // indicative of an old, upgradeable rig, and construct out the
780  // definition from there.
781  if (leapServiceProvider != null) {
782  firstLHCData = new OldLeapHandControllerData();
783  firstLHCData.lhcTransform = grandchild;
784  firstLHCData.leapServiceProvider = leapServiceProvider;
785 
786  firstLHCData.extraChildTransforms = new List<Transform>();
787  foreach (var greatGrandchild in grandchild.GetChildren()) {
788  firstLHCData.extraChildTransforms.Add(greatGrandchild);
789  }
790 
791  firstLHCData.handModelManager
792  = grandchild.GetComponent<HandModelManager>();
793

838  // (3) a LeapHandController transform

845  scannedRig.cameraData = null;

849  scannedRig.handModelParentTransform = null;

851  + "an old Leap rig.";

853  }

856  didDetectOldRig = true;
857

859  scannedRig.detectedAsPotentialRig = true;
860  scannedRig.leapSpaceData = firstLeapSpaceData;
861  scannedRig.lhcData = firstLHCData;
862  scannedRig.isUpgradeableOldRig = true;
863  scannedRig.reasonRigCannotBeUpgraded = "";

866  // parent transform. This will be the first child transform of the rig that
867  // contains one or more HandModelBase components.
868  // Not having one of these doesn't preclude updating the rig, since we can
869  // just make a new transform for this purpose.

871  .Query().Where(child =>

873  .Where(gc => gc.GetComponent<HandModelBase>() != null).Count() > 0)
874  .FirstOrDefault();

878  // old Leap image rig, with two eyes.
879  var leapImageRetriever = primaryCamData.cameraTransform
880  .GetComponent<LeapImageRetriever>();

882  if (leapImageRetriever != null) {
883  // If there is a secondary camera with a LeapEyeDislocator component, we'll
884  // assume that's the other (right) eye.
885  rightEyeCamData = secondaryCamDatas.Query()
886  .Where(cd => cd.cameraTransform.GetComponent<LeapEyeDislocator>() != null)
887  .FirstOrDefault();

891  scannedRig.imageRigData.rightEyeCamData = rightEyeCamData;
892  }
893  }
894  }
895  }

982

1007  }

1010  && cameraData.cameraComponent.stereoTargetEye == StereoTargetEyeMask.Left) {
1011  cameraData.cameraComponent.stereoTargetEye = StereoTargetEyeMask.Both;
1012  }
1013  }
1014 
1015  var cameraTransform = cameraData.cameraTransform;
1016 
1017  if (leapSpaceData == null) {
1018  Debug.LogError("No LeapSpace data available for this rig during upgrade "
