445  // Computes the angle to rotate the game object based on the change in the transform
446  //-------------------------------------------------
447  private void ComputeAngle( Hand hand )
448  {
449  Vector3 toHandProjected = ComputeToTransformProjected( hand.hoverSphereTransform );
450 
451  if ( !toHandProjected.Equals( lastHandProjected ) )
452  {
453  float absAngleDelta = Vector3.Angle( lastHandProjected, toHandProjected );
454 
455  if ( absAngleDelta > 0.0f )
456  {
457  if ( frozen )
458  {
459  float frozenSqDist = ( hand.hoverSphereTransform.position - frozenHandWorldPos ).sqrMagnitude;
460  if ( frozenSqDist > frozenSqDistanceMinMaxThreshold.x )
461  {
462  outAngle = frozenAngle + Random.Range( -1.0f, 1.0f );
463 
464  float magnitude = Util.RemapNumberClamped( frozenSqDist, frozenSqDistanceMinMaxThreshold.x, frozenSqDistanceMinMaxThreshold.y, 0.0f, 1.0f );
465  if ( magnitude > 0 )
466  {
467  StartCoroutine( HapticPulses( hand, magnitude, 10 ) );
468  }
469  else
470  {
471  StartCoroutine( HapticPulses( hand, 0.5f, 10 ) );
472  }
473 
474  if ( frozenSqDist >= frozenSqDistanceMinMaxThreshold.y )
475  {
476  onFrozenDistanceThreshold.Invoke();
477  }
478  }
479  }
480  else
481  {
482  Vector3 cross = Vector3.Cross( lastHandProjected, toHandProjected ).normalized;
483  float dot = Vector3.Dot( worldPlaneNormal, cross );
484 
485  float signedAngleDelta = absAngleDelta;
486 
487  if ( dot < 0.0f )
488  {
489  signedAngleDelta = -signedAngleDelta;
490  }
491 
492  if ( limited )
493  {
494  float angleTmp = Mathf.Clamp( outAngle + signedAngleDelta, minAngle, maxAngle );
495 
496  if ( outAngle == minAngle )
497  {
498  if ( angleTmp > minAngle && absAngleDelta < minMaxAngularThreshold )
499  {
500  outAngle = angleTmp;
501  lastHandProjected = toHandProjected;
502  }
503  }
504  else if ( outAngle == maxAngle )
505  {
506  if ( angleTmp < maxAngle && absAngleDelta < minMaxAngularThreshold )
507  {
508  outAngle = angleTmp;
509  lastHandProjected = toHandProjected;
510  }
511  }
512  else if ( angleTmp == minAngle )
513  {
514  outAngle = angleTmp;
515  lastHandProjected = toHandProjected;
516  onMinAngle.Invoke();
517  if ( freezeOnMin )
518  {
519  Freeze( hand );
520  }
521  }
522  else if ( angleTmp == maxAngle )
523  {
524  outAngle = angleTmp;
525  lastHandProjected = toHandProjected;
526  onMaxAngle.Invoke();
527  if ( freezeOnMax )
528  {
529  Freeze( hand );
530  }
531  }
532  else
533  {
534  outAngle = angleTmp;
535  lastHandProjected = toHandProjected;
536  }
537  }
538  else
539  {
540  outAngle += signedAngleDelta;
541  lastHandProjected = toHandProjected;
542  }
543  }
544  }
545  }
546  }
