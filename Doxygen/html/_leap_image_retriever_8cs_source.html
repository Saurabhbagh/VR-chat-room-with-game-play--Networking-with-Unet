160  private void addDistortionData(Image image, Color32[] colors, int startIndex) {
161  float[] distortionData = image.Distortion(Image.CameraType.LEFT).
162  Query().
163  Concat(image.Distortion(Image.CameraType.RIGHT)).
164  ToArray();
165 
166  for (int i = 0; i < distortionData.Length; i += 2) {
167  byte b0, b1, b2, b3;
168  encodeFloat(distortionData[i], out b0, out b1);
169  encodeFloat(distortionData[i + 1], out b2, out b3);
170  colors[i / 2 + startIndex] = new Color32(b0, b1, b2, b3);
171  }
172  }
173 
174  private void encodeFloat(float value, out byte byte0, out byte byte1) {
175  // The distortion range is -0.6 to +1.7. Normalize to range [0..1).
176  value = (value + 0.6f) / 2.3f;
177  float enc_0 = value;
178  float enc_1 = value * 255.0f;
179 
180  enc_0 = enc_0 - (int)enc_0;
181  enc_1 = enc_1 - (int)enc_1;
182 
183  enc_0 -= 1.0f / 255.0f * enc_1;
184 
185  byte0 = (byte)(enc_0 * 256.0f);
186  byte1 = (byte)(enc_1 * 256.0f);
187  }
188  }

273  private void LateUpdate() {
274  Frame imageFrame = _provider.CurrentFrame;
275 
276  _currentImage = null;
277 
278  /* Use the most recent image that is not newer than the current frame
279  * This means that the shown image might be slightly older than the current
280  * frame if for some reason a frame arrived before an image did.
281  *
282  * Usually however, this is just important when robust mode is enabled.
283  * At that time, image ids never line up with tracking ids.
284  */
285  Image potentialImage;
286  while (_imageQueue.TryPeek(out potentialImage)) {
287  if (potentialImage.SequenceId > imageFrame.Id) {
288  break;
289  }
290 
291  _currentImage = potentialImage;
292  _imageQueue.TryDequeue();
293  }
294  }
