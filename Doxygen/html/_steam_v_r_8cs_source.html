155  OpenVR.GetGenericInterface(OpenVR.IVRCompositor_Version, ref error);
156  if (error != EVRInitError.None)

162  return null;
163  }

166  if (error != EVRInitError.None)

168  initializedState = InitializedStates.InitializeFailure;
169  ReportError(error);
170  SteamVR_Events.Initialized.Send(false);

175 
176  if (Application.isEditor)
177  IdentifyEditorApplication();

180

184 
185 #if UNITY_EDITOR

615  private SteamVR()
616  {
617  hmd = OpenVR.System;
618  Debug.Log("<b>[SteamVR]</b> Initialized. Connected to " + hmd_TrackingSystemName + ":" + hmd_SerialNumber);
619 
620  compositor = OpenVR.Compositor;
621  overlay = OpenVR.Overlay;
622 
623  // Setup render values
624  uint w = 0, h = 0;
625  hmd.GetRecommendedRenderTargetSize(ref w, ref h);
626  sceneWidth = (float)w;
627  sceneHeight = (float)h;
628 
629  float l_left = 0.0f, l_right = 0.0f, l_top = 0.0f, l_bottom = 0.0f;
630  hmd.GetProjectionRaw(EVREye.Eye_Left, ref l_left, ref l_right, ref l_top, ref l_bottom);
631 
632  float r_left = 0.0f, r_right = 0.0f, r_top = 0.0f, r_bottom = 0.0f;
633  hmd.GetProjectionRaw(EVREye.Eye_Right, ref r_left, ref r_right, ref r_top, ref r_bottom);
634 
635  tanHalfFov = new Vector2(
636  Mathf.Max(-l_left, l_right, -r_left, r_right),
637  Mathf.Max(-l_top, l_bottom, -r_top, r_bottom));
638 
639  textureBounds = new VRTextureBounds_t[2];
640 
641  textureBounds[0].uMin = 0.5f + 0.5f * l_left / tanHalfFov.x;
642  textureBounds[0].uMax = 0.5f + 0.5f * l_right / tanHalfFov.x;
643  textureBounds[0].vMin = 0.5f - 0.5f * l_bottom / tanHalfFov.y;
644  textureBounds[0].vMax = 0.5f - 0.5f * l_top / tanHalfFov.y;
645 
646  textureBounds[1].uMin = 0.5f + 0.5f * r_left / tanHalfFov.x;
