60  // The target color for the Interaction object will be determined by various simple state checks.
61  Color targetColor = defaultColor;
62 
63  // "Primary hover" is a special kind of hover state that an InteractionBehaviour can
64  // only have if an InteractionHand's thumb, index, or middle finger is closer to it
65  // than any other interaction object.
66  if (_intObj.isPrimaryHovered && usePrimaryHover) {
67  targetColor = primaryHoverColor;
68  }
69  else {
70  // Of course, any number of objects can be hovered by any number of InteractionHands.
71  // InteractionBehaviour provides an API for accessing various interaction-related
72  // state information such as the closest hand that is hovering nearby, if the object
73  // is hovered at all.
74  if (_intObj.isHovered && useHover) {
75  float glow = _intObj.closestHoveringControllerDistance.Map(0F, 0.2F, 1F, 0.0F);
76  targetColor = Color.Lerp(defaultColor, hoverColor, glow);
77  }
78  }
79 
80  if (_intObj.isSuspended) {
81  // If the object is held by only one hand and that holding hand stops tracking, the
82  // object is "suspended." InteractionBehaviour provides suspension callbacks if you'd
83  // like the object to, for example, disappear, when the object is suspended.
84  // Alternatively you can check "isSuspended" at any time.

86  }

91  targetColor = pressedColor;
92  }
93
