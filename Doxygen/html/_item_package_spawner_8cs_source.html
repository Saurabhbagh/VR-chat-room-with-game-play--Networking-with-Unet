259  private void RemoveMatchingItemsFromHandStack( ItemPackage package, Hand hand )
260  {

299  private void SpawnAndAttachObject( Hand hand, GrabTypes grabType )
300  {
301  if ( hand.otherHand != null )
302  {
303  //If the other hand has this item package, take it back from the other hand
304  ItemPackage otherHandItemPackage = GetAttachedItemPackage( hand.otherHand );
305  if ( otherHandItemPackage == itemPackage )
306  {
307  TakeBackItem( hand.otherHand );
308  }
309  }
310 
311  if ( showTriggerHint )
312  {
313  hand.HideGrabHint();
314  }
315 
316  if ( itemPackage.otherHandItemPrefab != null )
317  {
318  if ( hand.otherHand.hoverLocked )
319  {
320  Debug.Log( "<b>[SteamVR Interaction]</b> Not attaching objects because other hand is hoverlocked and we can't deliver both items." );
321  return;
322  }
323  }
324 
325  // if we're trying to spawn a one-handed item, remove one and two-handed items from this hand and two-handed items from both hands
326  if ( itemPackage.packageType == ItemPackage.ItemPackageType.OneHanded )
327  {
328  RemoveMatchingItemTypesFromHand( ItemPackage.ItemPackageType.OneHanded, hand );
329  RemoveMatchingItemTypesFromHand( ItemPackage.ItemPackageType.TwoHanded, hand );
330  RemoveMatchingItemTypesFromHand( ItemPackage.ItemPackageType.TwoHanded, hand.otherHand );
331  }
332 
333  // if we're trying to spawn a two-handed item, remove one and two-handed items from both hands
334  if ( itemPackage.packageType == ItemPackage.ItemPackageType.TwoHanded )
335  {
336  RemoveMatchingItemTypesFromHand( ItemPackage.ItemPackageType.OneHanded, hand );
337  RemoveMatchingItemTypesFromHand( ItemPackage.ItemPackageType.OneHanded, hand.otherHand );
338  RemoveMatchingItemTypesFromHand( ItemPackage.ItemPackageType.TwoHanded, hand );
339  RemoveMatchingItemTypesFromHand( ItemPackage.ItemPackageType.TwoHanded, hand.otherHand );
340  }
341 
342  spawnedItem = GameObject.Instantiate( itemPackage.itemPrefab );
343  spawnedItem.SetActive( true );
344  hand.AttachObject( spawnedItem, grabType, attachmentFlags );
345 
346  if ( ( itemPackage.otherHandItemPrefab != null ) && ( hand.otherHand.isActive ) )
347  {
348  GameObject otherHandObjectToAttach = GameObject.Instantiate( itemPackage.otherHandItemPrefab );
349  otherHandObjectToAttach.SetActive( true );
350  hand.otherHand.AttachObject( otherHandObjectToAttach, grabType, attachmentFlags );
351  }
352 
353  itemIsSpawned = true;
354 
355  justPickedUpItem = true;
356 
357  if ( takeBackItem )
358  {
359  useFadedPreview = true;
360  pickupEvent.Invoke();
361  CreatePreviewObject();
362  }
363  }
364  }
