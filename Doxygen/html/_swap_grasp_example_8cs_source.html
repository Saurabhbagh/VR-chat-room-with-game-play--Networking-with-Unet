24  private bool _swapScheduled = false;

26  void Start() {
27  swapButton.OnUnpress += scheduleSwap;
28 
29  // Wait for just after the PhysX update to swap a grasp;
30  // this allows the swapped object to inherit the _latest_ rigidbody position and
31  // rotation from the original held object (which needs the PhysX update to receive
32  // scheduled force / MovePosition / MoveRotation changes from the grasped movement
33  // system).
34  PhysicsCallbacks.OnPostPhysics += onPostPhysics;
35  }
36 
37  private void scheduleSwap() {
38  _swapScheduled = true;
39  }
40 
41  private void onPostPhysics() {
42  //Swapping when both objects are grasped is unsupported
43  if(objA.isGrasped && objB.isGrasped) { return; }
44 
45  if (_swapScheduled && (objA.isGrasped || objB.isGrasped)) {
46 
47  // Swap "a" for "b"; a will be whichever object is the grasped one.
48  IntObj a, b;
49  if (objA.isGrasped) {
50  a = objA;
51  b = objB;
52  }
53  else {
54  a = objB;
55  b = objA;
56  }
57 
58  // (Optional) Remember B's pose and motion to apply to A post-swap.
59  var bPose = new Pose(b.rigidbody.position, b.rigidbody.rotation);
60  var bVel = b.rigidbody.velocity;
61  var bAngVel = b.rigidbody.angularVelocity;
62 
63  // Match the rigidbody pose of the originally held object before swapping.
64  // If it exists, always use the latestScheduledGraspPose to perform a SwapGrasp!
65  // This prevents subtle slippage with non-kinematic objects that may experience
66  // gravity forces, drag, or hit other objects, which can leak into the new
67  // grasping pose when the SwapGrasp is performed.
68  if (a.latestScheduledGraspPose.HasValue) {
69  b.rigidbody.position = a.latestScheduledGraspPose.Value.position;
70  b.rigidbody.rotation = a.latestScheduledGraspPose.Value.rotation;
71  }
72  else {
73  b.rigidbody.position = a.rigidbody.position;
74  b.rigidbody.rotation = a.rigidbody.rotation;
75  }
76 
77  // Swap!
78  a.graspingController.SwapGrasp(b);
79 
80  // Move A over to where B was, and for fun, let's give it B's motion as well.
81  a.rigidbody.position = bPose.position;
82  a.rigidbody.rotation = bPose.rotation;
83  a.rigidbody.velocity = bVel;
84  a.rigidbody.angularVelocity = bAngVel;
85  }
86 
87  _swapScheduled = false;
88  }
89  }
