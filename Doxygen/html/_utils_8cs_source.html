96  var _count = Pool<Dictionary<T, int>>.Spawn();
97  try {
98  int _nullCount = 0;
99 
100  foreach (var i in a) {
101  if (i == null) {
102  _nullCount++;
103  } else {
104  int count;
105  if (!_count.TryGetValue(i, out count)) {
106  count = 0;
107  }
108  _count[i] = count + 1;
109  }
110  }
111 
112  foreach (var i in b) {
113  if (i == null) {
114  _nullCount--;
115  } else {
116  int count;
117  if (!_count.TryGetValue(i, out count)) {
118  return false;
119  }
120  _count[i] = count - 1;
121  }
122  }
123 
124  if (_nullCount != 0) {
125  return false;
126  }
127 
128  foreach (var pair in _count) {
129  if (pair.Value != 0) {
130  return false;
131  }
132  }
133 
134  return true;
135  } finally {

278  public static string GenerateNiceName(string value) {
279  string result = "";
280  string curr = "";
281 
282  Func<char, bool> wordFunc = c => {
283  //Can't build any further if it's already capitalized
284  if (curr.Length > 0 && char.IsUpper(curr[0])) {
285  return false;
286  }
287 
288  //Can't add non-letters to words
289  if (!char.IsLetter(c)) {
290  return false;
291  }
292 
293  curr = c + curr;
294  return true;
295  };
296 
297  Func<char, bool> acronymFunc = c => {
298  //Can't add non-letters to acronyms
299  if (!char.IsLetter(c)) {
300  return false;
301  }
302 
303  //Can't add lowercase letters to acronyms
304  if (char.IsLower(c)) {
305  return false;
306  }
307 
308  curr = c + curr;
309  return true;
310  };
311 
312  Func<char, bool> numberFunc = c => {
313  //Can't add non-digits to a number
314  if (!char.IsDigit(c)) {
315  return false;
316  }
317 
318  curr = c + curr;
319  return true;
320  };
321 
322  Func<char, bool> fluffFunc = c => {
323  //Can't add digits or numbers to 'fluff'
324  if (char.IsDigit(c) || char.IsLetter(c)) {
325  return false;
326  }
327 
328  return true;
329  };
330 
331 
332  Func<char, bool> currFunc = null;
333  int currIndex = value.Length;
334 
335  while (currIndex != 0) {
336  currIndex--;
337  char c = value[currIndex];
338 
339  if (currFunc != null && currFunc(c)) {
340  continue;
341  }
342 
343  if (curr != "") {
344  result = " " + curr.Capitalize() + result;
345  curr = "";
346  }
347 
348  if (acronymFunc(c)) {
349  currFunc = acronymFunc;
350  } else if (wordFunc(c)) {
351  currFunc = wordFunc;
352  } else if (numberFunc(c)) {
353  currFunc = numberFunc;
354  } else if (fluffFunc(c)) {
355  currFunc = fluffFunc;
356  } else {
357  throw new Exception("Unexpected state, no function matched character " + c);
358  }
359  }
360 
361  if (curr != "") {
362  result = curr.Capitalize() + result;
363  }
364 
365  result = result.Trim();
366 
367  if (result.StartsWith("M ") || result.StartsWith("K ")) {
368  result = result.Substring(2);
369  }
370 
371  return result.Trim();
372  }

737  public static void FindOwnedChildComponents<ComponentType, OwnerType>
738  (OwnerType rootObj,
739  List<ComponentType> ownedComponents,
740  bool includeInactiveObjects = false)
741  where OwnerType : Component {
742  ownedComponents.Clear();
743  Stack<Transform> toVisit = Pool<Stack<Transform>>.Spawn();
744  List<ComponentType> componentsBuffer = Pool<List<ComponentType>>.Spawn();
745 
746  try {
747  toVisit.Push(rootObj.transform);
748  Transform curTransform;
749  while (toVisit.Count > 0) {
750  curTransform = toVisit.Pop();
751 
752  // Recursively search children and children's children.
753  foreach (var child in curTransform.GetChildren()) {
754  // Ignore children with OwnerType components of their own; its own OwnerType
755  // component owns its own ComponentType components and the ComponentType
756  // components of its children.
757  if (child.GetComponent<OwnerType>() == null
758  && (includeInactiveObjects || child.gameObject.activeInHierarchy)) {
759  toVisit.Push(child);
760  }
761  }
762 
763  // Since we'll visit every valid child, all we need to do is add the
764  // ComponentType components of every transform we visit.
765  componentsBuffer.Clear();
766  curTransform.GetComponents<ComponentType>(componentsBuffer);
767  foreach (var component in componentsBuffer) {
768  ownedComponents.Add(component);
769  }
770  }
771  } finally {
772  toVisit.Clear();
773  Pool<Stack<Transform>>.Recycle(toVisit);
774 
775  componentsBuffer.Clear();
776  Pool<List<ComponentType>>.Recycle(componentsBuffer);
777  }
778  }

927  public static void CompressQuatToBytes(Quaternion quat,
928  byte[] buffer,
929  ref int offset) {
930  int largest = 0;
931  float a, b, c;
932 
933  float abs_w = Mathf.Abs(quat.w);
934  float abs_x = Mathf.Abs(quat.x);
935  float abs_y = Mathf.Abs(quat.y);
936  float abs_z = Mathf.Abs(quat.z);
937 
938  float largest_value = abs_x;
939 
940  if (abs_y > largest_value) {
941  largest = 1;
942  largest_value = abs_y;
943  }
944  if (abs_z > largest_value) {
945  largest = 2;
946  largest_value = abs_z;
947  }
948  if (abs_w > largest_value) {
949  largest = 3;
950  largest_value = abs_w;
951  }
952  if (quat[largest] >= 0f) {
953  a = quat[(largest + 1) % 4];
954  b = quat[(largest + 2) % 4];
955  c = quat[(largest + 3) % 4];
956  }
957  else {
958  a = -quat[(largest + 1) % 4];
959  b = -quat[(largest + 2) % 4];
960  c = -quat[(largest + 3) % 4];
961  }
962 
963  // serialize
964  const float minimum = -1.0f / 1.414214f; // note: 1.0f / sqrt(2)
965  const float maximum = +1.0f / 1.414214f;
966  const float delta = maximum - minimum;
967  const uint maxIntegerValue = (1 << 10) - 1; // 10 bits
968  const float maxIntegerValueF = (float)maxIntegerValue;
969  float normalizedValue;
970  uint integerValue;
971 
972  uint sentData = ((uint)largest) << 30;
973  // a
974  normalizedValue = Mathf.Clamp01((a - minimum) / delta);
975  integerValue = (uint)Mathf.Floor(normalizedValue * maxIntegerValueF + 0.5f);
976  sentData = sentData | ((integerValue & maxIntegerValue) << 20);
977  // b
978  normalizedValue = Mathf.Clamp01((b - minimum) / delta);
979  integerValue = (uint)Mathf.Floor(normalizedValue * maxIntegerValueF + 0.5f);
980  sentData = sentData | ((integerValue & maxIntegerValue) << 10);
981  // c
982  normalizedValue = Mathf.Clamp01((c - minimum) / delta);
983  integerValue = (uint)Mathf.Floor(normalizedValue * maxIntegerValueF + 0.5f);
984  sentData = sentData | (integerValue & maxIntegerValue);
985 
986  BitConverterNonAlloc.GetBytes(sentData, buffer, ref offset);
987  }

1005  int largest = (int)(readData >> 30);
1006  float a, b, c;
1007 
1008  const float minimum = -1.0f / 1.414214f; // note: 1.0f / sqrt(2)
1009  const float maximum = +1.0f / 1.414214f;
1010  const float delta = maximum - minimum;
1011  const uint maxIntegerValue = (1 << 10) - 1; // 10 bits
1012  const float maxIntegerValueF = (float)maxIntegerValue;
1013  uint integerValue;
1014  float normalizedValue;
1015  // a
1016  integerValue = (readData >> 20) & maxIntegerValue;
1017  normalizedValue = (float)integerValue / maxIntegerValueF;
1018  a = (normalizedValue * delta) + minimum;
1019  // b
1020  integerValue = (readData >> 10) & maxIntegerValue;
1021  normalizedValue = (float)integerValue / maxIntegerValueF;
1022  b = (normalizedValue * delta) + minimum;
1023  // c
1024  integerValue = readData & maxIntegerValue;
1025  normalizedValue = (float)integerValue / maxIntegerValueF;
1026  c = (normalizedValue * delta) + minimum;
1027 
1028  Quaternion value = Quaternion.identity;
1029  float d = Mathf.Sqrt(1f - a * a - b * b - c * c);
1030  value[largest] = d;
1031  value[(largest + 1) % 4] = a;
1032  value[(largest + 2) % 4] = b;
1033  value[(largest + 3) % 4] = c;
1034 
1035  return value;
1036  }

1180  public static void FindColliders<T>(GameObject obj, List<T> colliders,
1181  bool includeInactiveObjects = false)
1182  where T : Collider {
1183  colliders.Clear();
1184  Stack<Transform> toVisit = Pool<Stack<Transform>>.Spawn();
1185  List<T> collidersBuffer = Pool<List<T>>.Spawn();
1186 
1187  try {
1188  // Traverse the hierarchy of this object's transform to find
1189  // all of its Colliders.
1190  toVisit.Push(obj.transform);
1191  Transform curTransform;
1192  while (toVisit.Count > 0) {
1193  curTransform = toVisit.Pop();
1194 
1195  // Recursively search children and children's children
1196  foreach (var child in curTransform.GetChildren()) {
1197  // Ignore children with Rigidbodies of their own; its own Rigidbody
1198  // owns its own colliders and the colliders of its children
1199  if (child.GetComponent<Rigidbody>() == null
1200  && (includeInactiveObjects || child.gameObject.activeSelf)) {
1201  toVisit.Push(child);
1202  }
1203  }
1204 
1205  // Since we'll visit every valid child, all we need to do is add the colliders
1206  // of every transform we visit.
1207  collidersBuffer.Clear();
1208  curTransform.GetComponents<T>(collidersBuffer);
1209  foreach (var collider in collidersBuffer) {
1210  colliders.Add(collider);
1211  }
1212  }
1213  } finally {
1214  toVisit.Clear();
1215  Pool<Stack<Transform>>.Recycle(toVisit);
1216 
1217  collidersBuffer.Clear();
1218  Pool<List<T>>.Recycle(collidersBuffer);
1219  }
1220  }
