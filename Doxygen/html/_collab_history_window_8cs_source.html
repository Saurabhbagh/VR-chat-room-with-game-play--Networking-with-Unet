202  public void UpdateRevisions(IEnumerable<RevisionData> datas, string tip, int totalRevisions, int currentPage)
203  {
204  var elements = new List<VisualElement>();
205  var isFullDateObtained = false; // Has everything from this date been obtained?
206  m_HistoryItems.Clear();
207 
208  if (datas != null)
209  {
210  DateTime currentDate = DateTime.MinValue;
211  foreach (var data in datas)
212  {
213  if (data.timeStamp.Date != currentDate.Date)
214  {
215  elements.Add(new CollabHistoryRevisionLine(data.timeStamp, isFullDateObtained));
216  currentDate = data.timeStamp;
217  }
218 
219  var item = new CollabHistoryItem(data);
220  m_HistoryItems.Add(item);
221 
222  var container = new VisualElement();
223  container.style.flexDirection = FlexDirection.Row;
224  if (data.current)
225  {
226  isFullDateObtained = true;
227  container.AddToClassList("currentRevision");
228  container.AddToClassList("obtainedRevision");
229  }
230  else if (data.obtained)
231  {
232  container.AddToClassList("obtainedRevision");
233  }
234  else
235  {
236  container.AddToClassList("absentRevision");
237  }
238  // If we use the index as-is, the latest commit will become #1, but we want it to be last
239  container.Add(new CollabHistoryRevisionLine(data.index));
240  container.Add(item);
241  elements.Add(container);
242  }
243  }
244 
245  m_HistoryView.scrollOffset = new Vector2(0, 0);
246  m_Pager.totalItems = totalRevisions;
247  m_Pager.curPage = currentPage;
248  m_Pager.items = elements;
