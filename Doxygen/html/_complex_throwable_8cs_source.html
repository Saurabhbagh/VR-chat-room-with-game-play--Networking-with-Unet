96  private void PhysicsAttach( Hand hand, GrabTypes startingGrabType )
97  {
98  PhysicsDetach( hand );
99 
100  Rigidbody holdingBody = null;
101  Vector3 holdingPoint = Vector3.zero;
102 
103  // The hand should grab onto the nearest rigid body
104  float closestDistance = float.MaxValue;
105  for ( int i = 0; i < rigidBodies.Count; i++ )
106  {
107  float distance = Vector3.Distance( rigidBodies[i].worldCenterOfMass, hand.transform.position );
108  if ( distance < closestDistance )
109  {
110  holdingBody = rigidBodies[i];
111  closestDistance = distance;
112  }
113  }
114 
115  // Couldn't grab onto a body
116  if ( holdingBody == null )
117  return;
118 
119  // Create a fixed joint from the hand to the holding body
120  if ( attachMode == AttachMode.FixedJoint )
121  {
122  Rigidbody handRigidbody = Util.FindOrAddComponent<Rigidbody>( hand.gameObject );
123  handRigidbody.isKinematic = true;
124 
125  FixedJoint handJoint = hand.gameObject.AddComponent<FixedJoint>();
126  handJoint.connectedBody = holdingBody;
127  }
128 
129  // Don't let the hand interact with other things while it's holding us
130  hand.HoverLock( null );
131 
132  // Affix this point
133  Vector3 offset = hand.transform.position - holdingBody.worldCenterOfMass;
134  offset = Mathf.Min( offset.magnitude, 1.0f ) * offset.normalized;
135  holdingPoint = holdingBody.transform.InverseTransformPoint( holdingBody.worldCenterOfMass + offset );
136 
137  hand.AttachObject( this.gameObject, startingGrabType, attachmentFlags );
138 
139  // Update holding list
140  holdingHands.Add( hand );
141  holdingBodies.Add( holdingBody );
142  holdingPoints.Add( holdingPoint );
143  }
