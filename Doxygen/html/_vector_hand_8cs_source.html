63  public void Encode(Hand fromHand) {
64  isLeft = fromHand.IsLeft;
65  palmPos = fromHand.PalmPosition.ToVector3();
66  palmRot = fromHand.Rotation.ToQuaternion();
67 
68  int boneIdx = 0;
69  for (int i = 0; i < 5; i++) {
70  Vector3 baseMetacarpal = ToLocal(fromHand.Fingers[i].bones[0].PrevJoint.ToVector3(),
71  palmPos, palmRot);
72  jointPositions[boneIdx++] = baseMetacarpal;
73  for (int j = 0; j < 4; j++) {
74  Vector3 joint = ToLocal(fromHand.Fingers[i].bones[j].NextJoint.ToVector3(),
75  palmPos, palmRot);
76  jointPositions[boneIdx++] = joint;
77  }
78  }
79  }

81  public void Decode(Hand intoHand) {
82  int boneIdx = 0;
83  Vector3 prevJoint = Vector3.zero;
84  Vector3 nextJoint = Vector3.zero;
85  Quaternion boneRot = Quaternion.identity;
86 
87  // Fill fingers.
88  for (int fingerIdx = 0; fingerIdx < 5; fingerIdx++) {
89  for (int jointIdx = 0; jointIdx < 4; jointIdx++) {
90  boneIdx = fingerIdx * 4 + jointIdx;
91  prevJoint = jointPositions[fingerIdx * 5 + jointIdx];
92  nextJoint = jointPositions[fingerIdx * 5 + jointIdx + 1];
93 
94  if ((nextJoint - prevJoint).normalized == Vector3.zero) {
95  // Thumb "metacarpal" slot is an identity bone.
96  boneRot = Quaternion.identity;
97  }
98  else {
99  boneRot = Quaternion.LookRotation(
100  (nextJoint - prevJoint).normalized,
101  Vector3.Cross((nextJoint - prevJoint).normalized,
102  (fingerIdx == 0 ?
103  (isLeft ? -Vector3.up : Vector3.up)
104  : Vector3.right)));
105  }
106 
107  // Convert to world space from palm space.
108  nextJoint = ToWorld(nextJoint, palmPos, palmRot);
109  prevJoint = ToWorld(prevJoint, palmPos, palmRot);
110  boneRot = palmRot * boneRot;
111 
112  intoHand.GetBone(boneIdx).Fill(
113  prevJoint: prevJoint.ToVector(),
114  nextJoint: nextJoint.ToVector(),
115  center: ((nextJoint + prevJoint) / 2f).ToVector(),
116  direction: (palmRot * Vector3.forward).ToVector(),
117  length: (prevJoint - nextJoint).magnitude,
118  width: 0.01f,
119  type: (Bone.BoneType)jointIdx,
120  rotation: boneRot.ToLeapQuaternion());
121  }
122  intoHand.Fingers[fingerIdx].Fill(
123  frameId: -1,
124  handId: (isLeft ? 0 : 1),
125  fingerId: fingerIdx,
126  timeVisible: Time.time,
127  tipPosition: nextJoint.ToVector(),
128  direction: (boneRot * Vector3.forward).ToVector(),
129  width: 1f,
130  length: 1f,
131  isExtended: true,
132  type: (Finger.FingerType)fingerIdx);
133  }
134 
135  // Fill arm data.
136  intoHand.Arm.Fill(ToWorld(new Vector3(0f, 0f, -0.3f), palmPos, palmRot).ToVector(),
137  ToWorld(new Vector3(0f, 0f, -0.055f), palmPos, palmRot).ToVector(),
138  ToWorld(new Vector3(0f, 0f, -0.125f), palmPos, palmRot).ToVector(),
139  Vector.Zero,
140  0.3f,
141  0.05f,
142  (palmRot).ToLeapQuaternion());
143 
144  // Finally, fill hand data.
145  intoHand.Fill(frameID: -1,
146  id: (isLeft ? 0 : 1),
147  confidence: 1f,
148  grabStrength: 0.5f,
149  grabAngle: 100f,
150  pinchStrength: 0.5f,
151  pinchDistance: 50f,
152  palmWidth: 0.085f,
153  isLeft: isLeft,
154  timeVisible: 1f,
155  fingers: null /* already uploaded finger data */,
156  palmPosition: palmPos.ToVector(),
157  stabilizedPalmPosition: palmPos.ToVector(),
158  palmVelocity: Vector3.zero.ToVector(),
159  palmNormal: (palmRot * Vector3.down).ToVector(),
160  rotation: (palmRot.ToLeapQuaternion()),
161  direction: (palmRot * Vector3.forward).ToVector(),
162  wristPosition: ToWorld(new Vector3(0f, 0f, -0.055f),
163  palmPos,
164  palmRot).ToVector());
165  }

185  public void ReadBytes(byte[] bytes, ref int offset) {
186  if (bytes.Length - offset < numBytesRequired) {
187  throw new System.IndexOutOfRangeException(
188  "Not enough room to read bytes for VectorHand encoding starting at offset "
189  + offset + " for array of size " + bytes + "; need at least "
190  + numBytesRequired + " bytes from the offset position.");
191  }
192 
193  // Chirality.
194  isLeft = bytes[offset++] == 0x00;
195 
196  // Palm position and rotation.
197  for (int i = 0; i < 3; i++) {
198  palmPos[i] = Convert.ToSingle(
199  BitConverterNonAlloc.ToInt16(bytes, ref offset))
200  / 4096f;
201  }
202  palmRot = Utils.DecompressBytesToQuat(bytes, ref offset);
203 
204  // Palm-local bone joint positions.
205  for (int i = 0; i < NUM_JOINT_POSITIONS; i++) {
206  for (int j = 0; j < 3; j++) {
207  jointPositions[i][j] = VectorHandExtensions.ByteToFloat(bytes[offset++]);
208  }
209  }
210  }

219  public void FillBytes(byte[] bytesToFill, ref int offset) {
220  if (_backingJointPositions == null) {
221  throw new System.InvalidOperationException(
222  "Joint positions array is null. You must fill a VectorHand with data before "
223  + "you can use it to fill byte representations.");
224  }
225 
226  if (bytesToFill.Length - offset < numBytesRequired) {
227  throw new System.IndexOutOfRangeException(
228  "Not enough room to fill bytes for VectorHand encoding starting at offset "
229  + offset + " for array of size " + bytesToFill.Length + "; need at least "
230  + numBytesRequired + " bytes from the offset position.");
231  }
232 
233  // Chirality.
234  bytesToFill[offset++] = isLeft ? (byte)0x00 : (byte)0x01;
235 
236  // Palm position, each component compressed
237  for (int i = 0; i < 3; i++) {
238  BitConverterNonAlloc.GetBytes(Convert.ToInt16(palmPos[i] * 4096f),
239  bytesToFill,
240  ref offset);
241  }
242 
243  // Palm rotation.
244  Utils.CompressQuatToBytes(palmRot, bytesToFill, ref offset);
245 
246  // Joint positions.
247  for (int j = 0; j < NUM_JOINT_POSITIONS; j++) {
248  for (int i = 0; i < 3; i++) {
249  bytesToFill[offset++] =
250  VectorHandExtensions.FloatToByte(jointPositions[j][i]);
251  }
252  }
253  }
