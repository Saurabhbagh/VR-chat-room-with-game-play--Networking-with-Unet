152  private Vector3 selectedDirection (Vector3 tipPosition) {
153  switch (PointingType) {
154  case PointingType.RelativeToHorizon:
155  Quaternion cameraRot = Camera.main.transform.rotation;
156  float cameraYaw = cameraRot.eulerAngles.y;
157  Quaternion rotator = Quaternion.AngleAxis(cameraYaw, Vector3.up);
158  return rotator * PointingDirection;
159  case PointingType.RelativeToCamera:
160  return Camera.main.transform.TransformDirection(PointingDirection);
161  case PointingType.RelativeToWorld:
162  return PointingDirection;
163  case PointingType.AtTarget:
164  if (TargetObject != null)
165  return TargetObject.position - tipPosition;
166  else return Vector3.zero;
167  default:
168  return PointingDirection;
169  }
170  }
