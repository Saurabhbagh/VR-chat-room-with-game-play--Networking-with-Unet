102  private void makeAttachmentPointsToggle(string attachmentFlagName, Vector2 offCenterPosImgSpace) {
103  AttachmentPointFlags attachmentPoints = target.attachmentPoints;
104 
105  AttachmentPointFlags flag = (AttachmentPointFlags)System.Enum.Parse(typeof(AttachmentPointFlags), attachmentFlagName, true);
106 
107  if (EditorGUI.Toggle(makeToggleRect(_handTexRect.center
108  + new Vector2(offCenterPosImgSpace.x * _handTexRect.width,
109  offCenterPosImgSpace.y * _handTexRect.height)),
110  (attachmentPoints & flag) == flag)) {
111 
112  target.attachmentPoints = attachmentPoints | flag; // Set flag bit to 1.
113  }
114  else {
115  if (!wouldFlagDeletionDestroyData(target, flag)
116  || EditorUtility.DisplayDialog("Delete " + flag + " Attachment Point?",
117  "Deleting the " + flag + " attachment point will destroy "
118  + "its GameObject and any of its non-Attachment-Point children, "
119  + "and will remove any components attached to it.",
120  "Delete " + flag + " Attachment Point", "Cancel")) {
121 
122  target.attachmentPoints = attachmentPoints & (~flag); // Set flag bit to 0.
123  }
124  }
125  }

132  private static bool wouldFlagDeletionDestroyData(AttachmentHands target, AttachmentPointFlags flag) {
133  if (target.attachmentHands == null) return false;
134 
135  foreach (var attachmentHand in target.attachmentHands) {
136  var point = attachmentHand.GetBehaviourForPoint(flag);
137 
138  if (point == null) return false;
139  else {
140  // Data will be destroyed if this AttachmentPointBehaviour's Transform contains any children
141  // that are not themselves AttachmentPointBehaviours.
142  foreach (var child in point.transform.GetChildren()) {
143  if (child.GetComponent<AttachmentPointBehaviour>() == null) return true;
144  }
145 
146  // Data will be destroyed if this AttachmentPointBehaviour contains any components
147  // that aren't constructed automatically.
148  foreach (var component in point.gameObject.GetComponents<Component>()) {
149  if (component is Transform) continue;
150  if (component is AttachmentPointBehaviour) continue;
151 
152  return true;
153  }
154  }
155  }
156  return false;
157  }
