191  SkinnedMeshRenderer highlightSkinned = highlightSkinnedRenderers[skinnedIndex];
192 
193  if (existingSkinned != null && highlightSkinned != null && attachedToHand == false)
194  {
195  highlightSkinned.transform.position = existingSkinned.transform.position;

197  highlightSkinned.transform.localScale = existingSkinned.transform.lossyScale;
198  highlightSkinned.localBounds = existingSkinned.localBounds;

297  (hand.otherHand.currentAttachedObjectInfo.Value.interactable != null &&
298  hand.otherHand.currentAttachedObjectInfo.Value.interactable.activateActionSetOnAttach != this.activateActionSetOnAttach))
299  {
300  activateActionSetOnAttach.Deactivate(hand.handType);
301  }
302  }
303

306  onDetachedFromHand.Invoke(hand);
307  }
308

310  if (skeletonPoser != null)
311  {
312  if (hand.skeleton != null)
313  hand.skeleton.BlendToSkeleton(releasePoseBlendTime);
