106  float eyeOffset = eye == Camera.StereoscopicEye.Left ? 1 : -1;

108  Vector3 providerForwardOffset = Vector3.zero,

110  Quaternion providerRotation = Quaternion.Euler(0f, 180f, 0f);
111  if (_provider is LeapXRServiceProvider) {
112  LeapXRServiceProvider _xrProvider = _provider as LeapXRServiceProvider;
113  providerForwardOffset = Vector3.forward * _xrProvider.deviceOffsetZAxis;
114  providerVerticalOffset = -Vector3.up * _xrProvider.deviceOffsetYAxis;
115  providerRotation = Quaternion.AngleAxis(_xrProvider.deviceTiltXAxis, Vector3.right);
116  } else {
