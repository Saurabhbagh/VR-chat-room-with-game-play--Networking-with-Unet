202  private static void processDragAndDrop(ISupportDragAndDrop dragAndDropSupport,
203  ref Rect r, SerializedProperty property) {
204  Event curEvent = Event.current;
205  Rect dropArea = dragAndDropSupport.GetDropArea(r, property);
206 
207  switch (curEvent.type) {
208  case EventType.Repaint:
209  case EventType.DragUpdated:
210  case EventType.DragPerform:
211  if (!dropArea.Contains(curEvent.mousePosition, allowInverse: true)) {
212  break;
213  }

216  DragAndDrop.objectReferences, property);
217 
218  if (isValidDrop) {

220  } else {

222  }

225  DragAndDrop.AcceptDrag();
