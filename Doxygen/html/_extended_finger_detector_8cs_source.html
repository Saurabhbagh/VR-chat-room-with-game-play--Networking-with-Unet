118  IEnumerator extendedFingerWatcher() {
119  Hand hand;
120  while(true){

123  hand = HandModel.GetLeapHand();
124  if(hand != null){

127  && matchFingerState(hand.Fingers[2], Middle)
128  && matchFingerState(hand.Fingers[3], Ring)
129  && matchFingerState(hand.Fingers[4], Pinky);
130 
131  int extendedCount = 0;
132  for (int f = 0; f < 5; f++) {

138  (extendedCount <= MaximumExtendedCount) &&
139  (extendedCount >= MinimumExtendedCount);

143  Deactivate();
